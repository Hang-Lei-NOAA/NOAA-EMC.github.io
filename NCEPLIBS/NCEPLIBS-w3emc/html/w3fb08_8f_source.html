2 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 C
4 C SUBPROGRAM: W3FB08 LAT/LON TO MERC (I,J) FOR GRIB
5 C PRGMMR: STACKPOLE ORG: NMC42 DATE:88-04-05
6 C
7 C ABSTRACT: CONVERTS A LOCATION ON EARTH GIVEN IN
8 C THE COORDINATE SYSTEM OF LATITUDE/LONGITUDE TO AN (I,J)
9 C COORDINATE SYSTEM OVERLAID ON A MERCATOR MAP PROJECTION
10 C W3FB08 IS THE REVERSE OF W3FB09
11 C USES GRIB SPECIFICATION OF THE LOCATION OF THE GRID
12 C
13 C PROGRAM HISTORY LOG:
14 C 88-03-01 ORIGINAL AUTHOR: STACKPOLE, W/NMC42
15 C 90-04-12 R.E.JONES CONVERT TO CRAY CFT77 FORTRAN
16 C
17 C USAGE: CALL W3FB08 (ALAT,ALON,ALAT1,ALON1,ALATIN,DX,XI,XJ)
18 C INPUT ARGUMENT LIST:
19 C ALAT - LATITUDE IN DEGREES (NEGATIVE IN SOUTHERN HEMIS)
20 C ALON - EAST LONGITUDE IN DEGREES, REAL*4
21 C ALAT1 - LATITUDE OF LOWER LEFT CORNER OF GRID (POINT (1,1))
22 C ALON1 - LONGITUDE OF LOWER LEFT CORNER OF GRID (POINT (1,1))
23 C ALL REAL*4
24 C ALATIN - THE LATITUDE AT WHICH THE MERCATOR CYLINDER
25 C INTERSECTS THE EARTH
26 C DX - MESH LENGTH OF GRID IN METERS AT ALATIN
27 C
28 C OUTPUT ARGUMENT LIST:
29 C XI - I COORDINATE OF THE POINT SPECIFIED BY ALAT, ALON
30 C XJ - J COORDINATE OF THE POINT; BOTH REAL*4
31 C
32 C REMARKS: FORMULAE AND NOTATION LOOSELY BASED ON HOKE, HAYES,
33 C AND RENNINGER'S "MAP PROJECTIONS AND GRID SYSTEMS...", MARCH 1981
34 C AFGWC/TN-79/003
35 C
36 C ATTRIBUTES:
37 C LANGUAGE: CRAY CFT77 FORTRAN
38 C MACHINE: CRAY Y-MP8/832
39 C
40 C$$$
41 C
42  DATA rerth /6.3712e+6/, pi/3.1416/
43 C
44 C PRELIMINARY VARIABLES AND REDIFINITIONS
45 C
46  radpd = pi / 180.0
47  degpr = 180.0 / pi
48  clain = cos(radpd*alatin)
49  dellon = dx / (rerth*clain)
50 C
51 C GET DISTANCE FROM EQUATOR TO ORIGIN ALAT1
52 C
53  djeo = 0.
54  IF (alat1.NE.0.)
55  & djeo = (alog(tan(0.5*((alat1+90.0)*radpd))))/dellon
56 C
57 C NOW THE I AND J COORDINATES
58 C
59  xi = 1. + ((alon - alon1)/(dellon*degpr))
60  xj = 1. + (alog(tan(0.5*((alat + 90.) * radpd))))/
61  & dellon
62  & - djeo
63 C
64  RETURN
65  END
