2 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 C
4 C SUBPROGRAM: W3FA06 CALCULATION OF THE LIFTED INDEX
5 C AUTHOR: HOWCROFT,J. ORG: W/NMC342 DATE: 78-07-01
6 C
7 C ABSTRACT: GIVEN THE PRESSURE,TEMPERATURE AND RELATIVE HUMIDITY OF
8 C AN AIR PARCEL AT SOME POINT IN THE ATMOSPHERE, CALCULATE THE
9 C LIFTED INDEX OF THE PARCEL. LIFTED INDEX IS DEFINED AS THE
10 C TEMPERATURE DIFFERENCE BETWEEN THE OBSERVED 500MB TEMPERATURE AND
11 C THE SUPPOSED TEMPERATURE THAT THE PARCEL WOULD OBTAIN IF IT WERE
12 C LIFTED DRY-ADIABATICALLY TO SATURATION AND THEN MOVED MOIST
13 C ADIABATICALLY TO THE 500MB LEVEL.
14 C
15 C PROGRAM HISTORY LOG:
16 C 78-07-01 J.HOWCROFT
17 C 89-01-24 R.E.JONES CHANGE TO MICROSOFT FORTRAN 4.10
18 C 90-06-08 R.E.JONES CHANGE TO SUN FORTRAN 1.3
19 C 91-03-29 R.E.JONES CONVERT TO SiliconGraphics FORTRAN
20 C 93-03-29 R.E.JONES ADD SAVE STATEMENT
21 C 95-09-25 R.E.JONES PUT IN W3 LIBRARY ON CRAY
22 C
23 C USAGE: CALL W3FA06(P,T,RH,T5,TLI)
24 C
25 C INPUT VARIABLES:
26 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES
27 C ------ --------- -----------------------------------------------
28 C P ARG LIST PARCEL PRESSURE IN MILLIBARS
29 C T ARG LIST PARCEL TEMPERATAURE IN DEGREES CELSIUS
30 C RH ARG LIST PARCEL RELATIVE HUMIDITY IN PERCENT
31 C T5 ARG LIST TEMPERATURE AT THE 500MB LEVEL IN DEG. CELSIUS
32 C
33 C OUTPUT VARIABLES:
34 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES
35 C ------ --------- -----------------------------------------------
36 C TLI ARG LIST LIFTED INDEX IN DEGREES CELSIUS
37 C TLI = 9.9999 ITERATION DIVERGES;
38 C RETURN TO USER PROGRAM
39 C
40 C SUBPROGRAMS CALLED:
41 C NAMES LIBRARY
42 C ------------------------------------------------------- --------

44 C W3FA01 W3LIB
45 C
46 C ATTRIBUTES:
47 C LANGUAGE: CRAY CFT77 FORTRAN
48 C MACHINE: CRAY C916/256, J916/2048

52  SAVE
53 C
54  DATA eps /0.5/
55  DATA kout / 6/
56 C
57  300 FORMAT (' *** ITERATION NOT CONVERGING IN W3FA06 ***')
58  350 FORMAT (' INPUT PARAMS ARE:',4f15.8,/
59  1 ' CALCULATIONS ARE',7e15.8)
60 C
61  potemp(t,p) = (t+273.16)*((1000./p)**0.2857)
62 C
63  eep(t,p,es) = exp((596.73-0.601*t)*((0.622*es)/(p-es))
64  1 / (0.24*(t+273.16)))
65 C
66  unpot(te,p) = (((p/1000.)**0.2857)*te)-273.16
67 C
68  vapres(t) = 6.11*exp(17.2694*t/(t+237.3))
69 C
70  CALL w3fa01 (p,t,rh,td,plcl,tlcl)
71  IF (plcl .GT. 500.) GO TO 30
72  IF (plcl .LT. 500.) GO TO 20
73  tli = t5 - tlcl
74  GO TO 80
75  20 CONTINUE
76 C LCL IS ABOVE THE 500MB LVL
77  tli = t5 - unpot((potemp(tlcl,plcl)),500.)
78  GO TO 80
79  30 CONTINUE
80 C USE STACKPOLE ALGORITHM (JAM VOL 6/1967 PP 464-7) TO FIND TGES
81 C SO THAT (TGES,500) IS ON SAME MOIST ADIABAT AS (TLCL,PLCL).
82  es = vapres(tlcl)
83  thd = potemp(tlcl,(plcl-es))
84  theta = thd * eep(tlcl,plcl,es)
85 C THETA IS THE PSEUDO-EQUIV POTENTIAL TEMP THRU (PLCL,TLCL).
86 C NOW FIND TEMP WHERE THETA INTERSECTS 500MB SFC.
87 C INITIALIZE FOR STACKPOLIAN ITERATION
88  tges = t5
89  dtt = 10.
90  piin = 1./(0.5**0.2857)
91  a = 0.
92  istp = 0
93 C START ITERATION.
94  40 CONTINUE
95  istp = istp + 1
96  IF (istp .GT. 200) GO TO 50
97  sva = vapres(tges)
98  ax = a
99  a = (tges+273.16)*piin * eep(tges,500.,sva) - theta
100  IF (abs(a) .LT. eps) GO TO 70
101  dtt = dtt * 0.5
102  IF (a*ax.LT.0.0) dtt = -dtt
103  tp = tges + dtt
104  sva = vapres(tp)
105  ap = (tp+273.16)*piin * eep(tp,500.,sva) - theta
106  IF (abs(ap) .LT. eps) GO TO 60
107 C FIND NEXT ESTIMATE, DTT IS ADJUSTMENT FROM OLD TO NEW TGES.
108  dtt = a*dtt/(a-ap)
109  IF (abs(dtt).LT.0.01) dtt = sign(0.01,dtt)
110  tges = tges + dtt
111  IF (tges .GT. 50) tges = 50.
112  GO TO 40
113 C
114  50 CONTINUE
115 C DISASTER SECTION
116  WRITE (kout,300)
117  WRITE (kout,350) p,t,rh,t5,theta,ax,a,ap,tges,tp,sva
118  tli = 9.9999
119  GO TO 80
120  60 CONTINUE
121  tges = tp
122  70 CONTINUE
123  tli = t5 - tges
124  80 CONTINUE
125  RETURN
126  END
