3 ! Module: mersenne_twister Modern random number generator
4 ! Prgmmr: Iredell Org: W/NX23 date: 2005-06-14
5 !
6 ! Abstract: This module calculates random numbers using the Mersenne twister.
7 ! (It has been adapted to a Fortran 90 module from open source software.
8 ! The comments from the original software are given below in the remarks.)
9 ! The Mersenne twister (aka MT19937) is a state-of-the-art random number
10 ! generator based on Mersenne primes and originally developed in 1997 by
11 ! Matsumoto and Nishimura. It has a period before repeating of 2^19937-1,
12 ! which certainly should be good enough for geophysical purposes. :-)
13 ! Considering the algorithm's robustness, it runs fairly speedily.
14 ! (Some timing statistics are given below in the remarks.)
15 ! This adaptation uses the standard Fortran 90 random number interface,
16 ! which can generate an arbitrary number of random numbers at one time.
17 ! The random numbers generated are uniformly distributed between 0 and 1.
18 ! The module also can generate random numbers from a Gaussian distribution
19 ! with mean 0 and standard deviation 1, using a Numerical Recipes algorithm.
20 ! The module also can generate uniformly random integer indices.
21 ! There are also thread-safe versions of the generators in this adaptation,
22 ! necessitating the passing of generator states which must be kept private.
23 !
24 ! Program History Log:
25 ! 2005-06-14 Mark Iredell
26 !
27 ! Usage:
28 ! The module can be compiled with 4-byte reals or with 8-byte reals, but
29 ! 4-byte integers are required. The module should be endian-independent.
30 ! The Fortran 90 interfaces random_seed and random_number are overloaded
31 ! and can be used as in the standard by adding the appropriate use statement
32 ! use mersenne_twister
33 ! In the below use cases, harvest is a real array of arbitrary size,
34 ! and iharvest is an integer array of arbitrary size.
35 ! To generate uniformly distributed random numbers between 0 and 1,
36 ! call random_number(harvest)
37 ! To generate Gaussian distributed random numbers with 0 mean and 1 sigma,
38 ! call random_gauss(harvest)
39 ! To generate uniformly distributed random integer indices between 0 and n,
40 ! call random_index(n,iharvest)
41 ! In standard "saved" mode, the random number generator can be used without
42 ! setting a seed. But to set a seed, only 1 non-zero integer is required, e.g.
43 ! call random_setseed(4357) ! set default seed
44 ! The full generator state can be set via the standard interface random_seed,
45 ! but it is recommended to use this method only to restore saved states, e.g.
46 ! call random_seed(size=lsave) ! get size of generator state seed array
47 ! allocate isave(lsave) ! allocate seed array
48 ! call random_seed(get=isave) ! fill seed array (then maybe save to disk)
49 ! call random_seed(put=isave) ! restore state (after read from disk maybe)
50 ! Locally kept generator states can also be saved in a seed array, e.g.
51 ! type(random_stat):: stat
52 ! call random_seed(get=isave,stat=stat) ! fill seed array
53 ! call random_seed(put=isave,stat=stat) ! restore state
54 ! To generate random numbers in a threaded region, the "thread-safe" mode
55 ! must be used where generator states of type random_state are passed, e.g.
56 ! type(random_stat):: stat(8)
57 ! do i=1,8 ! threadable loop
58 ! call random_setseed(7171*i,stat(i)) ! thread-safe call
59 ! enddo
60 ! do i=1,8 ! threadable loop
61 ! call random_number(harvest,stat(i)) ! thread-safe call
62 ! enddo
63 ! do i=1,8 ! threadable loop
64 ! call random_gauss(harvest,stat(i)) ! thread-safe call
65 ! enddo
66 ! do i=1,8 ! threadable loop
67 ! call random_index(n,iharvest,stat(i))! thread-safe call
68 ! enddo
69 ! There is also a relatively inefficient "interactive" mode available, where
70 ! setting seeds and generating random numbers are done in the same call.
71 ! There is also a functional mode available, returning one value at a time.
72 !

216 ! Sets and gets state; overloads Fortran 90 standard.
217  subroutine random_seed(size,put,get,stat)
218  implicit none
219  integer,intent(out),optional:: size
220  integer,intent(in),optional:: put(nrest)
221  integer,intent(out),optional:: get(nrest)
222  type(random_stat),intent(inout),optional:: stat
223  if(present(size)) then ! return size of seed array
224 ! if(present(put).or.present(get))&
225 ! call errmsg('RANDOM_SEED: more than one option set - some ignored')
226  size=nrest
227  elseif(present(put)) then ! restore from seed array
228 ! if(present(get))&
229 ! call errmsg('RANDOM_SEED: more than one option set - some ignored')
230  if(present(stat)) then
231  stat%mti=put(1)
232  stat%mt=put(2:n+1)
233  stat%iset=put(n+2)
234  stat%gset=transfer(put(n+3:nrest),stat%gset)
235  if(stat%mti.lt.0.or.stat%mti.gt.n.or.any(stat%mt.eq.0).or.
236  & stat%iset.lt.0.or.stat%iset.gt.1) then
237  call random_setseed_t(iseed,stat)
238 ! call errmsg('RANDOM_SEED: invalid seeds put - default seeds used')
239  endif
240  else
241  sstat%mti=put(1)
242  sstat%mt=put(2:n+1)
243  sstat%iset=put(n+2)
244  sstat%gset=transfer(put(n+3:nrest),sstat%gset)
245  if(sstat%mti.lt.0.or.sstat%mti.gt.n.or.any(sstat%mt.eq.0)
246  & .or.sstat%iset.lt.0.or.sstat%iset.gt.1) then
247  call random_setseed_t(iseed,sstat)
248 ! call errmsg('RANDOM_SEED: invalid seeds put - default seeds used')
249  endif
250  endif
251  elseif(present(get)) then ! save to seed array
252  if(present(stat)) then
253  if(stat%mti.eq.n+1) call random_setseed_t(iseed,stat)
254  get(1)=stat%mti
255  get(2:n+1)=stat%mt
256  get(n+2)=stat%iset
257  get(n+3:nrest)=transfer(stat%gset,get,nrest-(n+3)+1)
258  else
259  if(sstat%mti.eq.n+1) call random_setseed_t(iseed,sstat)
260  get(1)=sstat%mti
261  get(2:n+1)=sstat%mt
262  get(n+2)=sstat%iset
263  get(n+3:nrest)=transfer(sstat%gset,get,nrest-(n+3)+1)
264  endif
265  else ! reset default seed
266  if(present(stat)) then
267  call random_setseed_t(iseed,stat)
268  else
269  call random_setseed_t(iseed,sstat)
270  endif
271  endif
272  end subroutine

325 ! Subprogram random_number_t
326 ! Generates random numbers in thread-safe mode.
327  subroutine random_number_t(harvest,stat)
328  implicit none
329  real,intent(out):: harvest(:)
330  type(random_stat),intent(inout):: stat
331  integer j,kk,y
332  integer tshftu,tshfts,tshftt,tshftl
333  tshftu(y)=ishft(y,-11)
334  tshfts(y)=ishft(y,7)
335  tshftt(y)=ishft(y,15)
336  tshftl(y)=ishft(y,-18)
337  do j=1,size(harvest)
338  if(stat%mti.ge.n) then
339  do kk=0,n-m-1
340  y=ior(iand(stat%mt(kk),umask),iand(stat%mt(kk+1),lmask))
341  stat%mt(kk)=ieor(ieor(stat%mt(kk+m),ishft(y,-1)),
342  & mag01(iand(y,1)))
343  enddo
344  do kk=n-m,n-2
345  y=ior(iand(stat%mt(kk),umask),iand(stat%mt(kk+1),lmask))
346  stat%mt(kk)=ieor(ieor(stat%mt(kk+(m-n)),ishft(y,-1)),
347  & mag01(iand(y,1)))
348  enddo
349  y=ior(iand(stat%mt(n-1),umask),iand(stat%mt(0),lmask))
350  stat%mt(n-1)=ieor(ieor(stat%mt(m-1),ishft(y,-1)),
351  & mag01(iand(y,1)))
352  stat%mti=0
353  endif
354  y=stat%mt(stat%mti)
355  y=ieor(y,tshftu(y))
356  y=ieor(y,iand(tshfts(y),tmaskb))
357  y=ieor(y,iand(tshftt(y),tmaskc))
358  y=ieor(y,tshftl(y))
359  if(y.lt.0) then
360  harvest(j)=(real(y)+2.0**32)/(2.0**32-1.0)
361  else
362  harvest(j)=real(y)/(2.0**32-1.0)
363  endif
364  stat%mti=stat%mti+1
365  enddo
366  end subroutine

395 ! Subprogram random_gauss_t
396 ! Generates Gaussian random numbers in thread-safe mode.
397  subroutine random_gauss_t(harvest,stat)
398  implicit none
399  real,intent(out):: harvest(:)
400  type(random_stat),intent(inout):: stat
401  integer mx,my,mz,j
402  real r2(2),r,g1,g2
403  mz=size(harvest)
404  if(mz.le.0) return
405  mx=0
406  if(stat%iset.eq.1) then
407  mx=1
408  harvest(1)=stat%gset
409  stat%iset=0
410  endif
411  my=(mz-mx)/2*2+mx
412  do
413  call random_number_t(harvest(mx+1:my),stat)
414  do j=mx,my-2,2
415  call rgauss(harvest(j+1),harvest(j+2),r,g1,g2)
416  if(r.lt.1.) then
417  harvest(mx+1)=g1
418  harvest(mx+2)=g2
419  mx=mx+2
420  endif
421  enddo
422  if(mx.eq.my) exit
423  enddo
424  if(my.lt.mz) then
425  do
426  call random_number_t(r2,stat)
427  call rgauss(r2(1),r2(2),r,g1,g2)
428  if(r.lt.1.) exit
429  enddo
430  harvest(mz)=g1
431  stat%gset=g2
432  stat%iset=1
433  endif
434  contains
435 ! Numerical Recipes algorithm to generate Gaussian random numbers.
436  subroutine rgauss(r1,r2,r,g1,g2)
437  real,intent(in):: r1,r2
438  real,intent(out):: r,g1,g2
439  real v1,v2,fac
440  v1=2.*r1-1.
441  v2=2.*r2-1.
442  r=v1**2+v2**2
443  if(r.lt.1.) then
444  fac=sqrt(-2.*log(r)/r)
445  g1=v1*fac
446  g2=v2*fac
447  endif
448  end subroutine
449  end subroutine
