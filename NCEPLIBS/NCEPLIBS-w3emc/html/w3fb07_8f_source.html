1  SUBROUTINE w3fb07(XI,XJ,ALAT1,ALON1,DX,ALONV,ALAT,ALON)
2 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 C
4 C SUBPROGRAM: W3FB07 GRID COORDS TO LAT/LON FOR GRIB
5 C PRGMMR: STACKPOLE ORG: NMC42 DATE:88-04-05
6 C
7 C ABSTRACT: CONVERTS THE COORDINATES OF A LOCATION ON EARTH GIVEN IN A
8 C GRID COORDINATE SYSTEM OVERLAID ON A POLAR STEREOGRAPHIC MAP PRO-
9 C JECTION TRUE AT 60 DEGREES N OR S LATITUDE TO THE
10 C NATURAL COORDINATE SYSTEM OF LATITUDE/LONGITUDE
11 C W3FB07 IS THE REVERSE OF W3FB06.
12 C USES GRIB SPECIFICATION OF THE LOCATION OF THE GRID
13 C
14 C PROGRAM HISTORY LOG:
15 C 88-01-01 ORIGINAL AUTHOR: STACKPOLE, W/NMC42
16 C 90-04-12 R.E.JONES CONVERT TO CRAY CFT77 FORTRAN
17 C
18 C USAGE: CALL W3FB07(XI,XJ,ALAT1,ALON1,DX,ALONV,ALAT,ALON)
19 C INPUT ARGUMENT LIST:
20 C XI - I COORDINATE OF THE POINT REAL*4
21 C XJ - J COORDINATE OF THE POINT REAL*4
22 C ALAT1 - LATITUDE OF LOWER LEFT POINT OF GRID (POINT 1,1)
23 C LATITUDE <0 FOR SOUTHERN HEMISPHERE; REAL*4
24 C ALON1 - LONGITUDE OF LOWER LEFT POINT OF GRID (POINT 1,1)
25 C EAST LONGITUDE USED THROUGHOUT; REAL*4
26 C DX - MESH LENGTH OF GRID IN METERS AT 60 DEG LAT
27 C MUST BE SET NEGATIVE IF USING
28 C SOUTHERN HEMISPHERE PROJECTION; REAL*4
29 C 190500.0 LFM GRID,
30 C 381000.0 NH PE GRID, -381000.0 SH PE GRID, ETC.
31 C ALONV - THE ORIENTATION OF THE GRID. I.E.,
32 C THE EAST LONGITUDE VALUE OF THE VERTICAL MERIDIAN
33 C WHICH IS PARALLEL TO THE Y-AXIS (OR COLUMNS OF
34 C THE GRID) ALONG WHICH LATITUDE INCREASES AS
35 C THE Y-COORDINATE INCREASES. REAL*4
36 C FOR EXAMPLE:
37 C 255.0 FOR LFM GRID,
38 C 280.0 NH PE GRID, 100.0 SH PE GRID, ETC.
39 C
40 C OUTPUT ARGUMENT LIST:
41 C ALAT - LATITUDE IN DEGREES (NEGATIVE IN SOUTHERN HEMI.)
42 C ALON - EAST LONGITUDE IN DEGREES, REAL*4
43 C
44 C REMARKS: FORMULAE AND NOTATION LOOSELY BASED ON HOKE, HAYES,
45 C AND RENNINGER'S "MAP PROJECTIONS AND GRID SYSTEMS...", MARCH 1981
46 C AFGWC/TN-79/003
47 C
48 C ATTRIBUTES:
49 C LANGUAGE: CRAY CFT77 FORTRAN
50 C MACHINE: CRAY Y-MP8/832
51 C
52 C$$$
53 C
54  DATA rerth /6.3712e+6/,pi/3.1416/
55  DATA ss60 /1.86603/
56 C
57 C PRELIMINARY VARIABLES AND REDIFINITIONS
58 C
59 C H = 1 FOR NORTHERN HEMISPHERE; = -1 FOR SOUTHERN
60 C
61 C REFLON IS LONGITUDE UPON WHICH THE POSITIVE X-COORDINATE
62 C DRAWN THROUGH THE POLE AND TO THE RIGHT LIES
63 C ROTATED AROUND FROM ORIENTATION (Y-COORDINATE) LONGITUDE
64 C DIFFERENTLY IN EACH HEMISPHERE
65 C
66  IF (dx.LT.0) THEN
67  h = -1.0
68  dxl = -dx
69  reflon = alonv - 90.0
70  ELSE
71  h = 1.0
72  dxl = dx
73  reflon = alonv - 270.0
74  ENDIF
75 C
76  radpd = pi / 180.0
77  degprd = 180.0 / pi
78  rebydx = rerth / dxl
79 C
80 C RADIUS TO LOWER LEFT HAND (LL) CORNER
81 C
82  ala1 = alat1 * radpd
83  rmll = rebydx * cos(ala1) * ss60/(1. + h * sin(ala1))
84 C
85 C USE LL POINT INFO TO LOCATE POLE POINT
86 C
87  alo1 = (alon1 - reflon) * radpd
88  polei = 1. - rmll * cos(alo1)
89  polej = 1. - h * rmll * sin(alo1)
90 C
91 C RADIUS TO THE I,J POINT (IN GRID UNITS)
92 C
93  xx = xi - polei
94  yy = (xj - polej) * h
95  r2 = xx**2 + yy**2
96 C
97 C NOW THE MAGIC FORMULAE
98 C
99  IF (r2.EQ.0) THEN
100  alat = h * 90.
101  alon = reflon
102  ELSE
103  gi2 = (rebydx * ss60)**2
104  alat = degprd * h * asin((gi2 - r2)/(gi2 + r2))
105  arccos = acos(xx/sqrt(r2))
106  IF (yy.GT.0) THEN
107  alon = reflon + degprd * arccos
108  ELSE
109  alon = reflon - degprd * arccos
110  ENDIF
111  ENDIF
112  IF (alon.LT.0) alon = alon + 360.
113 C
114  RETURN
115  END
