3 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
4 ! . . . .
5 ! SUBPROGRAM: INSTRUMENT MONITOR WALL-CLOCK TIMES, ETC.
6 ! PRGMMR: IREDELL ORG: NP23 DATE:1998-07-16
7 !
8 ! ABSTRACT: THIS SUBPROGRAM IS USEFUL IN INSTRUMENTING A CODE
9 ! BY MONITORING THE NUMBER OF TIMES EACH GIVEN SECTION
10 ! OF A PROGRAM IS INVOKED AS WELL AS THE MINIMUM, MAXIMUM
11 ! AND TOTAL WALL-CLOCK TIME SPENT IN THE GIVEN SECTION.
12 !
13 ! PROGRAM HISTORY LOG:
14 ! 1998-07-16 IREDELL
15 ! 2019-09-17 FRIMEL and KALINA
16 ! DECOMPOSE RETURN STATISTCS IF STATEMENT
17 ! 2020-04-02 BOI VUONG
18 ! CHECK FOR KA > 0 AND MODIFILED IFBLOCK STATEMENT
19 ! INTO TWO SEPARATE IFBLOCK STATEMENTS
20 !
21 ! USAGE: CALL INSTRUMENT(K,KALL,TTOT,TMIN,TMAX)
22 ! INPUT ARGUMENT LIST:
23 ! K - INTEGER POSITIVE SECTION NUMBER
24 ! OR MAXIMUM SECTION NUMBER IN THE FIRST INVOCATION
25 ! OR ZERO TO RESET ALL WALL-CLOCK STATISTICS
26 ! OR NEGATIVE SECTION NUMBER TO SKIP MONITORING
27 ! AND JUST RETURN STATISTICS.
28 !
29 ! OUTPUT ARGUMENT LIST:
30 ! KALL - INTEGER NUMBER OF TIMES SECTION IS CALLED
31 ! TTOT - REAL TOTAL SECONDS SPENT IN SECTION
32 ! TMIN - REAL MINIMUM SECONDS SPENT IN SECTION
33 ! TMAX - REAL MAXIMUM SECONDS SPENT IN SECTION
34 !
35 ! SUBPROGRAMS CALLED:
36 ! W3UTCDAT RETURN THE UTC DATE AND TIME
37 ! W3DIFDAT RETURN A TIME INTERVAL BETWEEN TWO DATES
38 !
39 ! REMARKS:
40 ! THIS SUBPROGRAM SHOULD NOT BE INVOKED FROM A MULTITASKING REGION.
41 ! NORMALLY, TIME SPENT INSIDE THIS SUBPROGRAM IS NOT COUNTED.
42 ! WALL-CLOCK TIMES ARE KEPT TO THE NEAREST MILLISECOND.
43 !
44 ! EXAMPLE.
45 ! CALL INSTRUMENT(2,KALL,TTOT,TMIN,TMAX) ! KEEP STATS FOR 2 SUBS
46 ! DO K=1,N
47 ! CALL SUB1
48 ! CALL INSTRUMENT(1,KALL,TTOT,TMIN,TMAX) ! ACCUM STATS FOR SUB1
49 ! CALL SUB2
50 ! CALL INSTRUMENT(2,KALL,TTOT,TMIN,TMAX) ! ACCUM STATS FOR SUB2
51 ! ENDDO
52 ! PRINT *,'SUB2 STATS: ',KALL,TTOT,TMIN,TMAX
53 ! CALL INSTRUMENT(-1,KALL,TTOT,TMIN,TMAX) ! RETURN STATS FOR SUB1
54 ! PRINT *,'SUB1 STATS: ',KALL,TTOT,TMIN,TMAX
55 !
56 ! ATTRIBUTES:
57 ! LANGUAGE: FORTRAN 90
58 !
59 !$$$
60  IMPLICIT NONE
61  INTEGER,INTENT(IN):: K
62  INTEGER,INTENT(OUT):: KALL
63  REAL,INTENT(OUT):: TTOT,TMIN,TMAX
64  INTEGER,SAVE:: KMAX=0
65  INTEGER,DIMENSION(:),ALLOCATABLE,SAVE:: KALLS
66  REAL,DIMENSION(:),ALLOCATABLE,SAVE:: TTOTS,TMINS,TMAXS
67  INTEGER,DIMENSION(8),SAVE:: IDAT
68  INTEGER,DIMENSION(8):: JDAT
69  REAL,DIMENSION(5):: RINC
70  INTEGER:: KA
71 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
72  ka=abs(k)
73 ! ALLOCATE MONITORING ARRAYS IF INITIAL INVOCATION
74  IF(kmax.EQ.0) THEN
75  kmax=k
76  ALLOCATE(kalls(kmax))
77  ALLOCATE(ttots(kmax))
78  ALLOCATE(tmins(kmax))
79  ALLOCATE(tmaxs(kmax))
80  kalls=0
81  ka=0
82 ! OR RESET ALL STATISTICS BACK TO ZERO
83  ELSEIF(k.EQ.0) THEN
84  kalls=0
85 ! OR COUNT TIME SINCE LAST INVOCATION AGAINST THIS SECTION
86  ELSEIF(k.GT.0) THEN
87  CALL w3utcdat(jdat)
88  CALL w3difdat(jdat,idat,4,rinc)
89  kalls(k)=kalls(k)+1
90  IF(kalls(k).EQ.1) THEN
91  ttots(k)=rinc(4)
92  tmins(k)=rinc(4)
93  tmaxs(k)=rinc(4)
94  ELSE
95  ttots(k)=ttots(k)+rinc(4)
96  tmins(k)=min(tmins(k),rinc(4))
97  tmaxs(k)=max(tmaxs(k),rinc(4))
98  ENDIF
99  ENDIF
100 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
101 ! RETURN STATISTICS
102 
103 ! FRIMEL and KALINA, DECOMPOSE THE IF STATEMENT, SAFER FOR SOME
104 ! COMPILERS. Since No Guarantee on order of evaluation, and when
105 ! evaluation will stop.
106 ! MAKE SURE KA.GE.1 BEFORE TESTING IF KALLS(KA).GT.0, ELSE
107 ! MAY ENCOUNTER A RUNTIME SIGSEGV SEGEMENTATION FAULT.
108 ! Since Subscript #1 of the array KALLS can have value 0 which
109 ! is less than the lower bound of 1
110 ! IF(KA.GE.1.AND.KA.LE.KMAX.AND.KALLS(KA).GT.0) THEN
111 
112  IF(ka.GE.1.AND.ka.LE.kmax) THEN
113  IF(kalls(ka).GT.0) THEN
114  kall=kalls(ka)
115  ttot=ttots(ka)
116  tmin=tmins(ka)
117  tmax=tmaxs(ka)
118  ENDIF
119  IF(kalls(ka).LE.0) THEN
120  kall=0
121  ttot=0
122  tmin=0
123  tmax=0
124  ENDIF
125  END IF
126 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
127 ! KEEP CURRENT TIME FOR NEXT INVOCATION
128  IF(k.GE.0) CALL w3utcdat(idat)
129  END SUBROUTINE instrument
