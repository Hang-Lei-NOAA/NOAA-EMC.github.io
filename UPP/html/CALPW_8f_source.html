2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: CALPW COMPUTES
5 ! PRGRMMR: TREADON ORG: W/NP2 DATE: 92-12-24
6 !
7 ! ABSTRACT:
8 ! THIS ROUTINE COMPUTES PRECIPITABLE WATER IN A COLUMN
9 ! EXTENDING FROM THE FIRST ATMOSPHERIC ETA LAYER TO THE
10 ! MODEL TOP. THE DEFINITION USED IS
11 ! TOP
12 ! PRECIPITABLE WATER = SUM (Q+CLDW) DP*HTM/G
13 ! BOT
14 ! WHERE,
15 ! BOT IS THE FIRST ETA LAYER,
16 ! TOP IS THE MODEL TOP,
17 ! Q IS THE SPECIFIC HUMIDITY (KG/KG) IN THE LAYER
18 ! CLDW IS THE CLOUD WATER (KG/KG) IN THE LAYER
19 ! DP (Pa) IS THE LAYER THICKNESS.
20 ! HTM IS THE HEIGHT MASK AT THAT LAYER (=0 IF BELOW GROUND)
21 ! G IS THE GRAVITATIONAL CONSTANT
22 !
23 ! PROGRAM HISTORY LOG:
24 ! 92-12-24 RUSS TREADON
25 ! 96-03-04 MIKE BALDWIN - ADD CLOUD WATER AND SPEED UP CODE
26 ! 98-06-15 T BLACK - CONVERSION FROM 1-D TO 2-D
27 ! 00-01-04 JIM TUCCILLO - MPI VERSION
28 ! 02-06-19 MIKE BALDWIN - WRF VERSION
29 ! 04-12-30 H CHUANG - UPDATE TO CALCULATE TOTAL COLUMN FOR OTHER
30 ! HYDROMETEORS
31 ! 14-11-12 SARAH LU - UPDATE TO CALCULATE AEROSOL OPTICAL DEPTH
32 ! 15-07-02 SARAH LU - UPDATE TO CALCULATE SCATTERING AEROSOL
33 ! OPTICAL DEPTH (18)
34 ! 15-07-04 SARAH LU - CORRECT PW INTEGRATION FOR AOD (17)
35 ! 15-07-10 SARAH LU - UPDATE TO CALCULATE ASYMETRY PARAMETER
36 ! 19-07-25 Li(Kate) Zhang - MERGE SARHA LU's update for FV3-Chem
37 !
38 ! USAGE: CALL CALPW(PW)
39 ! INPUT ARGUMENT LIST:
40 ! PW - ARRAY OF PRECIPITABLE WATER.
41 !
42 ! OUTPUT ARGUMENT LIST:
43 ! NONE
44 !
45 ! OUTPUT FILES:

105  DO l = 1,lm

107 !$omp parallel do private(i,j)

118  ENDDO
119  ENDDO
120  ELSE IF (idecid == 3) THEN
121 !$omp parallel do private(i,j)

123  DO i=1,im
124  qdum(i,j) = qqi(i,j,l)

128 !$omp parallel do private(i,j)

146  ENDDO

149  ELSE IF (idecid == 16) THEN

175  qdum(i,j) = 0.

177  qdum(i,j) = qqi(i,j,l) + qqs(i,j,l)

179  ENDDO
180  ENDDO

182 ! SHORT WAVE T TENDENCY
183 !$omp parallel do private(i,j)

196  ENDDO

209  DO i=1,im

211  ENDDO
212  ENDDO

214 ! MOISTURE CONVERGENCE
215 !$omp parallel do private(i,j)

217  DO i=1,im
218  qdum(i,j) = mcvg(i,j,l)
219  ENDDO

222  ELSE IF (idecid == 14) THEN

261  ELSE IF (idecid == 19) THEN
262 !$omp parallel do private(i,j)

264  DO i=1,im
265  qdum(i,j) = taod5503d(i,j,l)

277

300  ENDDO
301  ENDDO
302  ENDDO ! l loop
303 
304 
305  IF (idecid == 14)THEN
306 !$omp parallel do private(i,j,dp)
307  DO j=jsta,jend
308  DO i=1,im
309  pw(i,j) = max(0.,pw(i,j)/pws(i,j)*100.)
310  ENDDO
311  ENDDO
312  END IF
313 ! convert ozone from kg/m2 to dobson units, which give the depth of the

315 
316  IF (idecid == 15) then
317 !$omp parallel do private(i,j)
318  DO j=jsta,jend
319  DO i=1,im
320  pw(i,j) = pw(i,j) / 2.14e-5
321  ENDDO
322  ENDDO
323  endif
324 !
325 ! END OF ROUTINE.
326 !
327  RETURN
328  END
