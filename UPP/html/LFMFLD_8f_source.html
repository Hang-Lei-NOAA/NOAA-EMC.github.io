1  SUBROUTINE lfmfld(RH3310,RH6610,RH3366,PW3310)
2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: LFMFLD COMPUTES LAYER MEAN LFM FIELDS
5 ! PRGRMMR: TREADON ORG: W/NP2 DATE: 92-12-22
6 !
7 ! ABSTRACT:
8 ! THIS ROUTINE COMPUTES THREE LAYER MEAN RELATIVE HUMIDITIES
9 ! AND A PRECIPITABLE WATER FIELD FROM ETA LEVEL DATA. THE
10 ! COMPUTED FIELDS ARE INTENDED TO MIMIC SIMILAR FIELDS COM-
11 ! PUTED BY THE LFM. THE ALGORITHM USED HERE IS FAIRLY PRI-
12 ! MATIVE. IN EACH COLUMN ABOVE A MASS POINT ON THE ETA GRID
13 ! WE SET THE FOLLOWING TARGET PRESSURES:
14 ! SIGMA LAYER 1.00 PRESSURE: SURFACE PRESSURE
15 ! SIGMA LAYER 0.66 PRESSURE: 0.50 * SURFACE PRESSURE
16 ! SIGMA LAYER 0.33 PRESSURE: 0.4356 * SURFACE PRESSURE
17 ! GIVEN THESE PRESSURES A SURFACE UP SUMMATION IS MADE OF
18 ! RELATIVE HUMIDITY AND/OR PRECIPITABLE WATER BETWEEN THESE
19 ! TARGET PRESSURES. EACH TERM IN THE SUMMATION IS WEIGHTED
20 ! BY THE THICKNESS OF THE ETA LAYER. THE FINAL LAYER MEAN
21 ! IS THIS SUM NORMALIZED BY THE TOTAL DEPTH OF THE LAYER.
22 ! THERE IS, OBVIOUSLY, NO NORMALIZATION FOR PRECIPITABLE WATER.
23 !
24 !
25 ! PROGRAM HISTORY LOG:
26 ! 92-12-22 RUSS TREADON
27 ! 93-07-27 RUSS TREADON - MODIFIED SUMMATION LIMITS FROM
28 ! 0.66*PSFC TO 0.75*PSFC AND 0.33*PSFC
29 ! TO 0.50*PSFC, WHERE PSFC IS THE
30 ! SURFACES PRESSURE. THE REASON FOR
31 ! THIS CHANGE WAS RECOGNITION THAT IN
32 ! THE LFM 0.33 AND 0.66 WERE MEASURED
33 ! FROM THE SURFACE TO THE TROPOPAUSE,
34 ! NOT THE TOP OF THE MODEL.
35 ! 93-09-13 RUSS TREADON - RH CALCULATIONS WERE MADE INTERNAL
36 ! TO THE ROUTINE.
37 ! 96-03-04 MIKE BALDWIN - CHANGE PW CALC TO INCLUDE CLD WTR
38 ! 98-06-16 T BLACK - CONVERSION FROM 1-D TO 2-D
39 ! 98-08-17 MIKE BALDWIN - COMPUTE RH OVER ICE
40 ! 98-12-22 MIKE BALDWIN - BACK OUT RH OVER ICE
41 ! 00-01-04 JIM TUCCILLO - MPI VERSION
42 ! 02-04-24 MIKE BALDWIN - WRF VERSION
43 !
44 !
45 ! USAGE: CALL LFMFLD(RH3310,RH6610,RH3366,PW3310)
46 ! INPUT ARGUMENT LIST:
47 ! NONE
48 !
49 ! OUTPUT ARGUMENT LIST:
50 ! RH3310 - SIGMA LAYER 0.33-1.00 MEAN RELATIVE HUMIDITY.
51 ! RH6610 - SIGMA LAYER 0.66-1.00 MEAN RELATIVE HUMIDITY.
52 ! RH3366 - SIGMA LAYER 0.33-0.66 MEAN RELATIVE HUMIDITY.
53 ! PW3310 - SIGMA LAYER 0.33-1.00 PRECIPITABLE WATER.
54 !
55 ! OUTPUT FILES:
56 ! NONE
57 !
58 ! LIBRARY:
59 ! COMMON -

65 ! LANGUAGE: FORTRAN

106  z6610 = d00
107  z3366 = d00

110  p10 = pint(i,j,nint(lmh(i,j)))
111  p66 = 0.75*p10
112  p33 = 0.50*p10
113  llmh = nint(lmh(i,j))
114 !
115 ! ACCULMULATE RELATIVE HUMIDITIES AND PRECIPITABLE WATER.
116 !
117  DO 10 l = llmh,1,-1
118 !
119 ! GET P, Z, T, AND Q AT MIDPOINT OF ETA LAYER.
120  alpm = d50*(alpint(i,j,l)+alpint(i,j,l+1))
121  dz = zint(i,j,l)-zint(i,j,l+1)
122  dp = pint(i,j,l+1)-pint(i,j,l)
123  pm = exp(alpm)
124  tm = t(i,j,l)
125  qm = q(i,j,l)
126  qm = amax1(qm,d00)
127 !
128 ! QS=PQ0/PM*EXP(A2*(TM-A3)/(TM-A4))
129  IF(modelname == 'GFS')THEN
130  es = min(fpvsnew(tm),pm)
131  qs = con_eps*es/(pm+con_epsm1*es)
132  ELSE
133  qs=pq0/pm*exp(a2*(tm-a3)/(tm-a4))
134  END IF
135  rh = qm/qs
136  IF (rh.GT.h1) THEN
137  rh = h1
138  qm = rh*qs
139  ENDIF
140  IF (rh.LT.d01) THEN
141  rh = d01
142  qm = rh*qs
143  ENDIF
144 !
145 ! JUMP OUT OF THIS LOOP IF WE ARE ABOVE THE HIGHEST TARGET PRESSURE.
146  IF (pm.LE.p33) GOTO 20
147 !
148 ! 0.66-1.00 RELATIVE HUMIDITY.
149  IF ((pm.LE.p10).AND.(pm.GE.p66)) THEN
150  z6610 = z6610 + dz
151  rh6610(i,j) = rh6610(i,j) + rh*dz
152  ENDIF
153 !
154 ! 0.33-1.00 RELATIVE HUMIDITY AND PRECIPITABLE WATER.
155  IF ((pm.LE.p10).AND.(pm.GE.p33)) THEN
156  z3310 = z3310 + dz
157  rh3310(i,j)= rh3310(i,j)+rh*dz
158  pw3310(i,j)= pw3310(i,j)+(q(i,j,l)+cwm(i,j,l))*dp*gi
159  ENDIF
160 !
161 ! 0.33-0.66 RELATIVE HUMIDITY.
162  IF ((pm.LE.p66).AND.(pm.GE.p33)) THEN
163  z3366 = z3366 + dz
164  rh3366(i,j) = rh3366(i,j) + rh*dz
165  ENDIF
166 !
167  10 CONTINUE
168  20 CONTINUE
169 !
170 ! NORMALIZE TO GET MEAN RELATIVE HUMIDITIES. AT
171 ! ONE TIME WE DIVIDED PRECIPITABLE WATER BY DENSITY
172 ! TO GET THE EQUIVALENT WATER DEPTH IN METERS. NO MORE.
173  IF (z6610.GT.d00) THEN
174  rh6610(i,j) = rh6610(i,j)/z6610
175  ELSE
176  rh6610(i,j) = spval
177  ENDIF
178 !
179  IF (z3310.GT.d00) THEN
180  rh3310(i,j) = rh3310(i,j)/z3310
181  ELSE
182  rh3310(i,j) = spval
183  ENDIF
184 !
185  IF (z3366.GT.d00) THEN
186  rh3366(i,j) = rh3366(i,j)/z3366
187  ELSE
188  rh3366(i,j) = spval
189  ENDIF
190  30 CONTINUE
191 !
192 !
193 ! END OF ROUTINE.
194 !
195  RETURN
196  END
