66  subroutine calwxt_bourg_post(im,jm,jsta_2l,jend_2u,jsta,jend,lm,lp1, &
67  & iseed,g,pthresh, &
68  & t,q,pmid,pint,lmh,prec,zint,ptype,me)
69 ! use mersenne_twister, only: random_number
70  use mersenne_twister
71  implicit none
72 !
73 ! input:
74  integer,intent(in):: im,jm,jsta_2l,jend_2u,jsta,jend,lm,lp1,iseed,me
75  real,intent(in):: g,pthresh
76  real,intent(in), dimension(im,jsta_2l:jend_2u,lm) :: t, q, pmid
77  real,intent(in), dimension(im,jsta_2l:jend_2u,lp1) :: pint, zint
78  real,intent(in), dimension(im,jsta_2l:jend_2u) :: lmh, prec
79 !
80 ! output:
81 ! real,intent(out) :: ptype(im,jm)
82  integer,intent(out) :: ptype(im,jsta:jend)
83 !
84  integer i,j,ifrzl,iwrml,l,lhiwrm,lmhk,jlen
85  real pintk1,areane,tlmhk,areape,pintk2,surfw,area1,dzkl,psfck,r1,r2
86  real rn(im*jm*2)
87 !
88 ! initialize weather type array to zero (ie, off).
89 ! we do this since we want ptype to represent the
90 ! instantaneous weather type on return.
91  print *,'in calwxtbg, jsta,jend=',jsta,jend,' im=',im
92  print *,'in calwxtbg,me=',me,'iseed=',iseed
93 !
94 !$omp parallel do
95  do j=jsta,jend
96  do i=1,im
97  ptype(i,j) = 0
98  enddo
99  enddo
100 !
101  jlen = jend - jsta + 1
102  call random_setseed(iseed)
103  call random_number(rn)
104 ! call random_number(rn,iseed)
105 !
106 !!$omp parallel do &
107 ! & private(a,lmhk,tlmhk,iwrml,psfck,lhiwrm,pintk1,pintk2,area1, &
108 ! & areape,dzkl,surfw,r1,r2)
109  print *,'incalwxtbg, rn',maxval(rn),minval(rn)
110 
111  do j=jsta,jend
112  if(me==1)print *,'incalwxtbg, j=',j
113  do i=1,im
114  lmhk = min(nint(lmh(i,j)),lm)
115  psfck = pint(i,j,lmhk+1)
116 !
117  if (prec(i,j) <= pthresh) cycle ! skip this point if no precip this time step
118 
119 ! find the depth of the warm layer based at the surface
120 ! this will be the cut off point between computing
121 ! the surface based warm air and the warm air aloft
122 !
123  tlmhk = t(i,j,lmhk) ! lowest layer t
124  iwrml = lmhk + 1
125  if (tlmhk >= 273.15) then
126  do l = lmhk, 2, -1
127  if (t(i,j,l) >= 273.15 .and. t(i,j,l-1) < 273.15 .and. &
128  & iwrml == lmhk+1) iwrml = l
129  end do
130  end if
131 !
132 ! now find the highest above freezing level
133 !
134 ! gsm added 250 mb check to prevent stratospheric warming situations
135 ! from counting as warm layers aloft
136  lhiwrm = lmhk + 1
137  do l = lmhk, 1, -1
138  if (t(i,j,l) >= 273.15 .and. pmid(i,j,l) > 25000.) lhiwrm = l
139  end do
140 
141 ! energy variables
142 
143 ! surfw is the positive energy between ground and the first sub-freezing layer above ground
144 ! areane is the negative energy between ground and the highest layer above ground
145 ! that is above freezing
146 ! areape is the positive energy "aloft" which is the warm energy not based at the ground
147 ! (the total warm energy = surfw + areape)
148 !
149 ! pintk1 is the pressure at the bottom of the layer
150 ! pintk2 is the pressure at the top of the layer
151 ! dzkl is the thickness of the layer
152 ! ifrzl is a flag that tells us if we have hit a below freezing layer
153 !
154  pintk1 = psfck
155  ifrzl = 0
156  areane = 0.0
157  areape = 0.0
158  surfw = 0.0
159 
160  do l = lmhk, 1, -1
161  if (ifrzl == 0.and.t(i,j,l) <= 273.15) ifrzl = 1
162  pintk2 = pint(i,j,l)
163  dzkl = zint(i,j,l)-zint(i,j,l+1)
164  area1 = log(t(i,j,l)/273.15) * g * dzkl
165  if (t(i,j,l) >= 273.15.and. pmid(i,j,l) > 25000.) then
166  if (l < iwrml) areape = areape + area1
167  if (l >= iwrml) surfw = surfw + area1
168  else
169  if (l > lhiwrm) areane = areane + abs(area1)
170  end if
171  pintk1 = pintk2
172  end do
173 !
174 ! decision tree time
175 !
176  if (areape < 2.0) then
177 ! very little or no positive energy aloft, check for
178 ! positive energy just above the surface to determine rain vs. snow
179  if (surfw < 5.6) then
180 ! not enough positive energy just above the surface
181 ! snow = 1
182  ptype(i,j) = 1
183  else if (surfw > 13.2) then
184 ! enough positive energy just above the surface
185 ! rain = 8
186  ptype(i,j) = 8
187  else
188 ! transition zone, assume equally likely rain/snow
189 ! picking a random number, if <=0.5 snow
190  r1 = rn(i+im*(j-1))
191  if (r1 <= 0.5) then
192  ptype(i,j) = 1 ! snow = 1
193  else
194  ptype(i,j) = 8 ! rain = 8
195  end if
196  end if
197 !
198  else
199 ! some positive energy aloft, check for enough negative energy
200 ! to freeze and make ice pellets to determine ip vs. zr
201  if (areane > 66.0+0.66*areape) then
202 ! enough negative area to make ip,
203 ! now need to check if there is enough positive energy
204 ! just above the surface to melt ip to make rain
205  if (surfw < 5.6) then
206 ! not enough energy at the surface to melt ip
207  ptype(i,j) = 2 ! ice pellets = 2
208  else if (surfw > 13.2) then
209 ! enough energy at the surface to melt ip
210  ptype(i,j) = 8 ! rain = 8
211  else
212 ! transition zone, assume equally likely ip/rain
213 ! picking a random number, if <=0.5 ip
214  r1 = rn(i+im*(j-1))
215  if (r1 <= 0.5) then
216  ptype(i,j) = 2 ! ice pellets = 2
217  else
218  ptype(i,j) = 8 ! rain = 8
219  end if
220  end if
221  else if (areane < 46.0+0.66*areape) then
222 ! not enough negative energy to refreeze, check surface temp
223 ! to determine rain vs. zr
224  if (tlmhk < 273.15) then
225  ptype(i,j) = 4 ! freezing rain = 4
226  else
227  ptype(i,j) = 8 ! rain = 8
228  end if
229  else
230 ! transition zone, assume equally likely ip/zr
231 ! picking a random number, if <=0.5 ip
232  r1 = rn(i+im*(j-1))
233  if (r1 <= 0.5) then
234 ! still need to check positive energy
235 ! just above the surface to melt ip vs. rain
236  if (surfw < 5.6) then
237  ptype(i,j) = 2 ! ice pellets = 2
238  else if (surfw > 13.2) then
239  ptype(i,j) = 8 ! rain = 8
240  else
241 ! transition zone, assume equally likely ip/rain
242 ! picking a random number, if <=0.5 ip
243  r2 = rn(i+im*(j-1)+im*jm)
244  if (r2 <= 0.5) then
245  ptype(i,j) = 2 ! ice pellets = 2
246  else
247  ptype(i,j) = 8 ! rain = 8
248  end if
249  end if
250  else
251 ! not enough negative energy to refreeze, check surface temp
252 ! to determine rain vs. zr
253  if (tlmhk < 273.15) then
254  ptype(i,j) = 4 ! freezing rain = 4
255  else
256  ptype(i,j) = 8 ! rain = 8
257  end if
258  end if
259  end if
260  end if
261 ! write(1000+me,*)' finished for i, j, from calbourge me=',me,i,j
262  end do
263 ! write(1000+me,*)' finished for j, from calbourge me=',me,j
264  end do
265 ! write(1000+me,*)' returning from calbourge me=',me
266  return
267  end
