94 ! SUBROUTINE PROFQ2.F. OUTER LOOP OVER THE VERTICAL.

97 !$omp parallel do private(i,j,l,ie,iw,cs,ct,dthvkl,dukl,dvkl, &
98 !$omp& rdzkl,ri,uhkl,ulkl,vhkl,vlkl,wndsl,wndslp)

101  if(gridtype /= 'A')THEN

103  call exch(uh(1,jsta_2l,l))
104  call exch(vh(1,jsta_2l,l))

106  call exch(uh(1,jsta_2l,l+1))

112 !

129  wndslp = vtm(i,j-1,l+1) + vtm(iw,j,l+1)+ &

131  IF(wndsl == 0. .OR. wndslp == 0.) cycle
132  uhkl = (uh(i,j-1,l)+uh(iw,j,l)+uh(ie,j,l)+uh(i,j+1,l))/wndsl
133  ulkl = (uh(i,j-1,l+1)+uh(iw,j,l+1)+uh(ie,j,l+1)+ &

136  vlkl = (vh(i,j-1,l+1)+vh(iw,j,l+1)+vh(ie,j,l+1)+ &

138  ELSE IF(gridtype == 'B')THEN

143  uh(i,j,l+1))/4.0

147  END IF

149  rdzkl = 1.0 / (zmid(i,j,l)-zmid(i,j,l+1))

152 ! QROOT = SQRT(Q2KL)
153 ! ELKL = EL(I,J,L)
154 ! ELKL = MAX(ELKL,EPSQ2)
155 ! ELKLSQ = ELKL*ELKL

157  dthvkl = thv(i,j,l)-thv(i,j,l+1)
158  dukl = (uhkl-ulkl) * rdzkl
159  dvkl = (vhkl-vlkl) * rdzkl
160  cs = dukl*dukl + dvkl*dvkl
161 !
162 ! COMPUTE GRADIENT RICHARDSON NUMBER.
163 !

167 ! NUMBER TO POST PROCESSOR SPECIAL VALUE.

169  richno(i,j,l) = spval

172 !

174 !
175  ct = -1.*g*beta*dthvkl*rdzkl

177  richno(i,j,l) = ri
178  ENDIF
179 !
180  ENDDO
181  ENDDO
182  ENDDO ! end of l loop
