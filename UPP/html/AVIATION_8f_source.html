2  SUBROUTINE calllws(U,V,H,LLWS)
3 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
4 ! . . .
5 ! SUBPROGRAM: CALLLWS COMPUTES Low Level Wind Shear (0-2000feet)
6 ! PRGRMMR: Binbin Zhou /NCEP/EMC DATE: 2005-08-16
7 !
8 ! ABSTRACT:
9 ! This program computes the low level wind shear(LLWS) over 0-2000 feet (0-609.5m)
10 ! layer. But because 10m wind represent sfc wind, 10-619.5 m layer
11 ! is used. (NOAA/NWS Instruction 10-813, 2004)
12 !
13 ! Definition: LLWS(Z1,Z2) is vector difference of wind at z1 and z2
14 ! where Z1 = 10m + Surface height
15 ! Z2 = 619.5 + Surface height
16 !
17 ! Algorithm: since Z2 is not defined in the model, so,
18 ! first thing is searching Z2 to see which layers
19 ! it is located(ie between which two pressure levels),
20 ! then find the wind vector (U2,V2)at Z2 by interpolating with
21 ! the wind vectors of the at pressure levels above and below
22 ! then compute the vector difference between Z2 and Z1 (ie U10,V10)
23 !
24 !
25 !
26 !
27 ! ----------------------------------------- K2-1 ---------------------
28 ! ^
29 ! |
30 ! |
31 ! |
32 ! ____ | _____ Z2, U2=interpo[U(K2),U(K2-1)]
33 ! ^ | V2=interpo[V(K2),V(K2-1)]
34 ! | |
35 ! ------------|---------|------------------ K2 ------------------------
36 ! | |
37 ! | |DH=SUM of all layers between K1-1 & K2-1
38 ! | | .
39 ! |609.5m | .
40 ! |(2000ft) | .
41 ! | v
42 ! ------------|---------------------------------------------------- LSM-2
43 ! | ^
44 ! | |ZH1
45 ! | |
46 ! o-o 10m ___v__ Z1,U10,V10
47 ! FIS ....|..... ^
48 ! ^ . . |
49 ! --|-------------------------|------------ K1 -------------------- LSM-1
50 ! | . . |
51 ! |. . |
52 ! .| ... |
53 ! --|-------------------------|------------------------------------- LSM
54 ! . | |
55 ! ////////////////////////////////////////////////////////////////// Sea Level
56 !
57 !
58 ! USAGE: CALL CALLLWS(U,V,H,LLWS)
59 ! INPUT ARGUMENT LIST:
60 ! U - U wind profile (m/s) (at pressure level)
61 ! V - V wind (m/s) (at pressure level)
62 ! H - Height (m) (at pressure level)
63 !
64 ! OUTPUT ARGUMENT LIST:
65 ! LLWS - Low level wind shear (Knots/2000ft)
66 !
67 ! OUTPUT FILES:
68 ! NONE
69 !
70 ! SUBPROGRAMS CALLED:
71 ! UTILITIES:
72 ! LIBRARY:
73 ! NONE
74 !
75 ! ATTRIBUTES:
76 ! LANGUAGE: FORTRAN 90/77
77 ! MACHINE : BLUE AT NCEP
78 !$$$
79 !
80  USE vrbls2d, only: fis, u10, v10
81  use params_mod, only: gi
82  use ctlblk_mod, only: jsta, jend, im, jm, lsm
83 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
84  implicit none
85 !
86 ! DECLARE VARIABLES.
87 !
88  REAL,DIMENSION(IM,JM,LSM),INTENT(IN) :: U,V,H
89  REAL,DIMENSION(IM,JM),INTENT(INOUT) :: LLWS
90  REAL :: Z1,Z2,HZ1,DH,U2,V2,W2,RT
91  INTEGER :: K1,K2
92  integer I,J,LP
93 
94 !***************************************************************
95 !
96 !
97 
98  DO 100 j=jsta,jend
99  DO i=1,im
100 
101  z1 = 10.0 + fis(i,j)*gi !Height of 10m levels geographic height (from sea level)
102 
103  IF(z1.LT.h(i,j,lsm)) THEN !First search location of 10m wind level
104  k1 = lsm + 1 !to see it is in which pressure levels
105  ELSE
106  DO lp = lsm,2,-1 !If not found, keep searching upward
107  IF(z1.GE.h(i,j,lp).AND.z1.LT.h(i,j,lp-1)) THEN
108  k1 = lp
109  END IF
110  END DO
111  END IF
112 
113  hz1 = h(i,j,k1-1) - z1 !Distance between K1-1 and 10m level
114 
115  dh = 0.0
116 
117  IF((hz1+10).GT.609.6) THEN !Then, search 2000ft(609.6m) location
118  u2= u10(i,j) + (u(i,j,k1-1)-u10(i,j))*599.6/hz1 !found it between K1-1 and K1, then linear
119  v2= v10(i,j) + (v(i,j,k1-1)-v10(i,j))*599.6/hz1 !interpolate to get wind at 2000ft U2,V2
120  z2= fis(i,j)*gi + 609.6
121  ELSE !otherwise, keep on search upward
122  DO lp = k1-1,2,-1
123  dh=dh+(h(i,j,lp-1) - h(i,j,lp))
124  IF((dh+hz1+10).gt.609.6) THEN !found the 2000ft level
125  z2=fis(i,j)*gi+609.6
126  rt=(z2-h(i,j,lp))/(h(i,j,lp-1)-h(i,j,lp))
127  u2=u(i,j,lp)+rt*(u(i,j,lp-1)-u(i,j,lp))
128  v2=v(i,j,lp)+rt*(v(i,j,lp-1)-v(i,j,lp))
129  k2=lp
130  GO TO 610
131  END IF
132  END DO
133  END IF
134 
135 !computer vector difference
136 610 llws(i,j)=sqrt((u2-u10(i,j))**2+(v2-v10(i,j))**2)/ &
137  609.6 * 1.943*609.6 !unit: knot/2000ft
138 
139 
140  ENDDO
141 
142 100 CONTINUE
143 
144  RETURN
145  END

148  SUBROUTINE calicing (T1,RH,OMGA, ICING)
149 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
150 ! . . .
151 ! SUBPROGRAM: CALICING COMPUTES In-Flight Icing
152 ! PRGRMMR: Binbin Zhou /NCEP/EMC DATE: 2005-08-16
153 !
154 ! ABSTRACT:
155 ! This program computes the in-flight icing condition
156 ! with the T-RH-OMGA algorithm provided by S. Silberberg of
157 ! NCEP/AWC (improved new version)
158 !
159 ! According to S. Silberberg, Icing happens in following
160 ! situation:
161 ! (1) -22C < T < 0C to
162 ! (2) RH > 70 %
163 ! (3) Ascent air, OMGA < 0
164 ! (4) Equivalent Potential Vorticity (EPV) < 0
165 ! (5) Cloud water if SLD (supercooled large droplet)
166 !
167 ! Current version dosn't consider SLD, so cloud water
168 ! is not used. EPV computation is not available for current
169 ! NCEP/EMC models(NAM, WRF, RSM), so EPV is also not
170 ! used
171 !
172 ! USAGE: CALL CALICING(T1,RH,OMGA,ICING)
173 ! INPUT ARGUMENT LIST:
174 ! T1 - TEMPERATURE (K)
175 ! RH - RELATIVE HUMIDITY (DECIMAL FORM)
176 ! OMGA - Vertical velocity (Pa/sec)
177 !
178 ! OUTPUT ARGUMENT LIST:
179 ! ICING - ICING CONDITION (1 or 0)
180 !
181 ! OUTPUT FILES:
182 ! NONE
183 !
184 ! SUBPROGRAMS CALLED:
185 ! UTILITIES:
186 ! LIBRARY:
187 ! NONE
188 !
189 ! ATTRIBUTES:
190 ! LANGUAGE: FORTRAN 90/77
191 ! MACHINE : BLUE AT NCEP
192 !$$$
193 !
194  use ctlblk_mod, only: jsta, jend, im
195 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
196  implicit none
197 !
198 ! DECLARE VARIABLES.
199 !
200  REAL, DIMENSION(IM,jsta:jend), INTENT(IN) :: T1,RH,OMGA
201  REAL, DIMENSION(IM,jsta:jend), INTENT(INOUT) :: ICING
202  integer I,J
203 !***************************************************************
204 !
205 !
206  DO j=jsta,jend
207  DO i=1,im
208 
209  IF(omga(i,j) < 0.0 .AND. &
210  (t1(i,j) <= 273.0 .AND. t1(i,j) >= 251.0) &
211  .AND. rh(i,j) >= 70.0) THEN
212 
213  icing(i,j) = 1.0
214  ELSE
215  icing(i,j) = 0.0
216  END IF
217  ENDDO
218  ENDDO
219 
220  RETURN
221  END
