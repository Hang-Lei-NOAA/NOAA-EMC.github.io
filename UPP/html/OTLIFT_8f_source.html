2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: OTLIFT COMPUTES SFC TO 500MB LIFTED INDEX
5 ! PRGRMMR: TREADON ORG: W/NP2 DATE: 93-03-10
6 !
7 ! ABSTRACT:
8 ! THIS ROUTINE COMPUTES A SURFACE TO 500MB LIFTED INDEX.
9 ! THE LIFTED PARCEL IS FROM THE FIRST ATMOSPHERIC ETA
10 ! LAYER (IE, THE ETA LAYER CLOSEST TO THE MODEL GROUND).
11 ! THE LIFTED INDEX IS THE DIFFERENCE BETWEEN THIS PARCEL'S
12 ! TEMPERATURE AT 500MB AND THE AMBIENT 500MB TEMPERATURE.
13 ! .
14 !
15 ! PROGRAM HISTORY LOG:
16 ! ??-??-?? ??? - SUBROUTINE OTLIFT IN ETA MODEL.
17 ! 93-03-10 RUSS TREADON - ADAPTED OTLIFT FOR USE WITH NEW POST.
18 ! 98-06-18 T BLACK - CONVERSION FROM 1-D TO 2-D
19 ! 00-01-04 JIM TUCCILLO - MPI VERSION
20 ! 01-10-25 H CHUANG - MODIFIED TO PROCESS HYBRID MODEL OUTPUT
21 ! 02-06-11 MIKE BALDWIN - WRF VERSION
22 ! 11-04-12 GEOFF MANIKIN - USE VIRTUAL TEMPERATURE
23 !
24 ! USAGE: CALL OTLIFT(SLINDX)
25 ! INPUT ARGUMENT LIST:
26 !
27 ! OUTPUT ARGUMENT LIST:
28 ! SLINDX - LIFTED INDEX.
29 !
30 ! OUTPUT FILES:
31 ! NONE
32 !
33 ! SUBPROGRAMS CALLED:
34 ! UTILITIES:
35 ! NONE
36 ! LIBRARY:
37 ! COMMON - CTLBLK
38 !

40 ! LANGUAGE: FORTRAN
41 ! MACHINE : CRAY C-90
42 !$$$

44  use vrbls3d, only: pmid, t, q
45  use vrbls2d, only: t500
46  use masks, only: lmh
47  use lookup_mod, only: thl, rdth, jtb, qs0, sqs, rdq,itb, ptbl, pl, &
48  rdp, the0, sthe, rdthe, ttbl
49  use ctlblk_mod, only: jsta, jend, im
50  use params_mod, only: d00,h10e5, capa, elocp, eps, oneps
51 !
52 
53 !
54  implicit none
55 !
56 ! SET LOCAL PARAMETERS.
57  real,PARAMETER :: D8202=.820231e0 , h5e4=5.e4 , p500=50000.
58  real,external::FPVSNEW
59 
60 !
61 ! DECLARE VARIABLES.
62  real,intent(out) :: SLINDX(IM,jsta:jend)
63  REAL :: TVP, ESATP, QSATP
64  REAL :: TTH, TP, APESP, PARTMP, THESP, TPSP
65  REAL :: BQS00, SQS00, BQS10, SQS10, BQ, SQ, TQ
66  REAL :: P00, P10, P01, P11, T00, T10, T01, T11
67  REAL :: BTHE00, STHE00, BTHE10, STHE10, BTH, STH
68  REAL :: TQQ, QQ, QBT, TTHBT, TBT, APEBT, PPQ, PP
69 !
70  INTEGER :: I, J, LBTM, ITTBK, IQ, IT, IPTBK, ITH, IP, IQTB
71  INTEGER :: ITTB, IPTB, ITHTB
72 !
73 !***********************************************************************
74 ! START OTLIFT HERE
75 !
76 ! INTIALIZE LIFTED INDEX ARRAY TO ZERO.
77 !$omp parallel do private(i,j)
78  DO j=jsta,jend
79  DO i=1,im
80  slindx(i,j) = d00
81  ENDDO
82  ENDDO
83 !--------------FIND EXNER AT LOWEST LEVEL-------------------------------
84  DO j=jsta,jend
85  DO i=1,im
86  lbtm=nint(lmh(i,j))
87  tbt = t(i,j,lbtm)
88  qbt = q(i,j,lbtm)
89  apebt = (h10e5/pmid(i,j,lbtm))**capa
90 !--------------SCALING POTENTIAL TEMPERATURE & TABLE INDEX--------------
91  tthbt = tbt*apebt
92  tth = (tthbt-thl)*rdth
93  tqq = tth-aint(tth)
94  ittb = int(tth)+1
95 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
96  IF(ittb .LT. 1)THEN
97  ittb = 1
98  tqq = d00
99  ENDIF
100  IF(ittb .GE. jtb)THEN
101  ittb = jtb-1
102  tqq = d00
103  ENDIF
104 !--------------BASE AND SCALING FACTOR FOR SPEC. HUMIDITY---------------
105  ittbk = ittb
106  bqs00=qs0(ittbk)
107  sqs00=sqs(ittbk)
108  bqs10=qs0(ittbk+1)
109  sqs10=sqs(ittbk+1)
110 !--------------SCALING SPEC. HUMIDITY & TABLE INDEX---------------------
111  bq=(bqs10-bqs00)*tqq+bqs00
112  sq=(sqs10-sqs00)*tqq+sqs00
113  tq=(qbt-bq)/sq*rdq
114  ppq = tq-aint(tq)
115  iqtb = int(tq)+1
116 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
117  IF(iqtb .LT. 1)THEN
118  iqtb = 1
119  ppq = d00
120  ENDIF
121  IF(iqtb .GE. itb)THEN
122  iqtb = itb-1
123  ppq = d00
124  ENDIF
125 !--------------SATURATION PRESSURE AT FOUR SURROUNDING TABLE PTS.-------
126  iq=iqtb
127  it = ittb
128  p00=ptbl(iq,it)
129  p10=ptbl(iq+1,it)
130  p01=ptbl(iq,it+1)
131  p11=ptbl(iq+1,it+1)
132 !--------------SATURATION POINT VARIABLES AT THE BOTTOM-----------------
133  tpsp = p00+(p10-p00)*ppq+(p01-p00)*tqq &
134  +(p00-p10-p01+p11)*ppq*tqq
135  IF(tpsp .LE. d00) tpsp = h10e5
136  apesp = (h10e5/tpsp)**capa
137  thesp = tthbt*exp(elocp*qbt*apesp/tthbt)
138 !--------------SCALING PRESSURE & TT TABLE INDEX------------------------
139  tp=(h5e4-pl)*rdp
140  qq = tp-aint(tp)
141  iptb = int(tp)+1
142 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
143  IF(iptb .LT. 1)THEN
144  iptb = 1
145  qq = d00
146  ENDIF
147  IF(iptb .GE. itb)THEN
148  iptb = itb-1
149  qq = d00
150  ENDIF
151 !--------------BASE AND SCALING FACTOR FOR THE--------------------------
152  iptbk=iptb
153  bthe00=the0(iptbk)
154  sthe00=sthe(iptbk)
155  bthe10=the0(iptbk+1)
156  sthe10=sthe(iptbk+1)
157 !--------------SCALING THE & TT TABLE INDEX-----------------------------
158  bth=(bthe10-bthe00)*qq+bthe00
159  sth=(sthe10-sthe00)*qq+sthe00
160  tth=(thesp-bth)/sth*rdthe
161  pp = tth-aint(tth)
162  ithtb = int(tth)+1
163 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
164  IF(ithtb .LT. 1)THEN
165  ithtb = 1
166  pp = d00
167  ENDIF
168  IF(ithtb .GE. jtb)THEN
169  ithtb = jtb-1
170  pp = d00
171  ENDIF
172 !--------------TEMPERATURE AT FOUR SURROUNDING TT TABLE PTS.------------
173  ith=ithtb
174  ip=iptb
175  t00=ttbl(ith,ip)
176  t10=ttbl(ith+1,ip)
177  t01=ttbl(ith,ip+1)
178  t11=ttbl(ith+1,ip+1)
179 !--------------PARCEL TEMPERATURE AT 500MB----------------------------
180  IF(tpsp .GE. h5e4)THEN
181  partmp=(t00+(t10-t00)*pp+(t01-t00)*qq &
182  +(t00-t10-t01+t11)*pp*qq)
183  ELSE
184  partmp=tbt*apebt*d8202
185  ENDIF
186 !--------------LIFTED INDEX---------------------------------------------
187 !
188 ! GSM THE PARCEL TEMPERATURE AT 500 MB HAS BEEN COMPUTED, AND WE
189 ! FIND THE MIXING RATIO AT THAT LEVEL WHICH WILL BE THE SATURATION
190 ! VALUE SINCE WE'RE FOLLOWING A MOIST ADIABAT. NOTE THAT THE
191 ! AMBIENT 500 MB SHOULD PROBABLY BE VIRTUALIZED, BUT THE IMPACT
192 ! OF MOISTURE AT THAT LEVEL IS QUITE SMALL
193 
194  esatp=fpvsnew(partmp)
195  qsatp=eps*esatp/(p500-esatp*oneps)
196  tvp=partmp*(1+0.608*qsatp)
197  slindx(i,j)=t500(i,j)-tvp
198  ENDDO
199  ENDDO
200 ! write(*,*) ' in otlift t500 partmp ',t500(1,1),partmp(1,1)
201 ! write(*,*) ' in otlift tbt ',tbt(1,1)
202 !
203  RETURN
204  END
