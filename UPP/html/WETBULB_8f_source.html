11 !-----------------------------------------------------------------------
12 ! ROUTINE TO COMPUTE WET BULB TEMPERATURES USING THE LOOK UP TABLE
13 ! APPROACH THAT IS USED IN CUCNVC
14 !
15 ! FOR A GIVEN POINT K AND LAYER L:
16 ! THETA E IS COMPUTED FROM THETA AND Q BY LIFTING THE PARCEL TO
17 ! ITS SATURATION POINT.
18 ! THEN THE WET BULB TEMPERATURE IS FOUND BY FOLLOWING THE THETA E
19 ! DOWN TO THE ORIGINAL PRESSURE LEVEL (USING SUBROUTINE TTBLEX).
20 !
21 !
22  use lookup_mod, only: thl, rdth, jtb, qs0, sqs, rdq, itb, ptbl, plq, ttbl,&
23  pl, rdp, the0, sthe, rdthe, ttblq, itbq, jtbq, rdpq, the0q, stheq,&
24  rdtheq
25  use ctlblk_mod, only: jsta, jend, im, jsta_2l, jend_2u, lm
26  use cuparm_mod, only: h10e5, capa, epsq, d00, elocp
27 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
28  implicit none
29 !
30 !-----------------------------------------------------------------------
31 ! LIST OF VARIABLES NEEDED
32 ! PARAMETERS:
33 ! INCLUDED IN "cuparm" AND "parm.tbl"
34 ! INPUT:
35 ! T,Q,HTM,PMID(3-D),KARR (2-D)
36 ! OUTPUT:
37 ! TWET (3-D)
38 ! SUBROUTINES CALLED:
39 ! TTBLEX
40 !
41  real,dimension(IM,jsta_2l:jend_2u,LM),intent(in) :: T,Q, &
42  pmid,htm
43  integer,dimension(IM,jsta:jend), intent(in) :: KARR
44  real,dimension(IM,jsta_2l:jend_2u,LM),intent(out) :: TWET
45 
46 
47  real, dimension(im,jsta:jend) :: THESP, QQ, PP
48  integer, dimension(im,jsta:jend) :: KLRES,KHRES,IPTB,ITHTB
49 !
50  integer I,J,L,ITTB1,ITTBK,IQTBK,IT,KNUML,KNUMH,IQ
51  real TBTK,QBTK,APEBTK,TTHBTK,TTHK,QQK,BQS00K,SQS00K,BQS10K, &
52  sqs10k,bqk,sqk,tqk,ppk,tpspk,apespk,prespk,p00k,p10k,p01k, &
53  p11k,presk
54 !
55 !--------------COMPUTE WET BULB TEMPERATURES----------------------------
56 !!$omp parallel do
57 !!$omp& private(apebtk,apespk,bqk,bqs00k,bqs10k,iq,iqtbk,it,ittb1,ittbk,
58 !!$omp& karr,khres,klres,knumh,knuml,p00k,p01k,p10k,p11k,ppk,
59 !!$omp& presk,qbtk,qqk,sqk,sqs00k,sqs10k,tbtk,thesp,tpspk,
60 !!$omp& tqk,tthbtk,tthk)
61 !-----------------------------------------------------------------------
62  DO 300 l=1,lm
63  DO 125 j=jsta,jend
64  DO 125 i=1,im
65  IF (htm(i,j,l).LT.1.0) THEN
66  thesp(i,j)=273.15
67  GOTO 125
68  ENDIF
69  tbtk =t(i,j,l)
70  qbtk =q(i,j,l)
71  presk =pmid(i,j,l)
72  apebtk=(h10e5/presk)**capa
73  IF(qbtk.LT.epsq) qbtk=htm(i,j,l)*epsq
74 !--------------SCALING POTENTIAL TEMPERATURE & TABLE INDEX--------------
75  tthbtk =tbtk*apebtk
76  tthk =(tthbtk-thl)*rdth
77  qqk =tthk-aint(tthk)
78  ittb1 =int(tthk)+1
79 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
80  IF(ittb1.LT.1) THEN
81  ittb1 =1
82  qqk =d00
83  ENDIF
84 !
85  IF(ittb1.GE.jtb) THEN
86  ittb1 =jtb-1
87  qqk =d00
88  ENDIF
89 !--------------BASE AND SCALING FACTOR FOR SPEC. HUMIDITY---------------
90  ittbk=ittb1
91  bqs00k=qs0(ittbk)
92  sqs00k=sqs(ittbk)
93  bqs10k=qs0(ittbk+1)
94  sqs10k=sqs(ittbk+1)
95 !--------------SCALING SPEC. HUMIDITY & TABLE INDEX---------------------
96  bqk=(bqs10k-bqs00k)*qqk+bqs00k
97  sqk=(sqs10k-sqs00k)*qqk+sqs00k
98  tqk=(qbtk-bqk)/sqk*rdq
99  ppk=tqk-aint(tqk)
100  iqtbk=int(tqk)+1
101 !--------------KEEPING INDICES WITHIN THE TABLE-------------------------
102  IF(iqtbk.LT.1) THEN
103  iqtbk =1
104  ppk =d00
105  ENDIF
106 !
107  IF(iqtbk.GE.itb) THEN
108  iqtbk=itb-1
109  ppk =d00
110  ENDIF
111 !--------------SATURATION PRESSURE AT FOUR SURROUNDING TABLE PTS.-------
112  iq=iqtbk
113  it=ittb1
114  p00k=ptbl(iq ,it )
115  p10k=ptbl(iq+1,it )
116  p01k=ptbl(iq ,it+1)
117  p11k=ptbl(iq+1,it+1)
118 !--------------SATURATION POINT VARIABLES AT THE BOTTOM-----------------
119  tpspk=p00k+(p10k-p00k)*ppk+(p01k-p00k)*qqk &
120  +(p00k-p10k-p01k+p11k)*ppk*qqk
121  apespk=(h10e5/tpspk)**capa
122  thesp(i,j)=tthbtk*exp(elocp*qbtk*apespk/tthbtk)
123 ! ENDIF
124  125 CONTINUE
125 !--------------SCALING PRESSURE & TT TABLE INDEX------------------------
126  knuml=0
127  knumh=0
128 !
129  DO 280 j=jsta,jend
130  DO 280 i=1,im
131  klres(i,j)=0
132  khres(i,j)=0
133 !
134 ! IF(KARR(I,J).GT.0)THEN
135  presk=pmid(i,j,l)
136 !
137  IF(presk.LT.plq)THEN
138  knuml=knuml+1
139  klres(i,j)=1
140  ELSE
141  knumh=knumh+1
142  khres(i,j)=1
143  ENDIF
144 ! ENDIF
145  280 CONTINUE
146 !***
147 !*** COMPUTE PARCEL TEMPERATURE ALONG MOIST ADIABAT FOR PRESSURE<PL
148 !**
149  IF(knuml.GT.0)THEN
150  CALL ttblex(twet(1,jsta_2l,l),ttbl,itb,jtb,klres &
151  ,pmid(1,jsta_2l,l),pl,qq,pp,rdp,the0,sthe &
152  ,rdthe,thesp,iptb,ithtb)
153  ENDIF
154 !***
155 !*** COMPUTE PARCEL TEMPERATURE ALONG MOIST ADIABAT FOR PRESSURE>PL
156 !**
157  IF(knumh.GT.0)THEN
158  CALL ttblex(twet(1,jsta_2l,l),ttblq,itbq,jtbq,khres &
159  ,pmid(1,jsta_2l,l),plq,qq,pp,rdpq,the0q,stheq &
160  ,rdtheq,thesp,iptb,ithtb)
161  ENDIF
162 !-----------------------------------------------------------------------
163 !-----------------------------------------------------------------------
164  300 CONTINUE
165  RETURN
166  END
