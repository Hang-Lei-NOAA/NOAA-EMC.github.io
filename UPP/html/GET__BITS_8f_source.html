93  SUBROUTINE fndbit ( rmin, rmax, rdb, nmbts, iscale, iret )
94 !************************************************************************
95 !* FNDBIT *
96 !* *
97 !* This subroutine computes the number of packing bits given the *
98 !* maximum number of significant digits to preserve or the binary *
99 !* precision to store the data. The binary precision is given as a *
100 !* negative integer, ISCALE will always be zero in this case. *
101 !* *
102 !* The binary precision translates as follows: *
103 !* -1 => store data to nearest 1/2 *
104 !* -2 => store data to nearest 1/4 *
105 !* -3 => store data to nearest 1/8 *
106 !* *
107 !* Note that a fractional number of significant digits is allowed. *
108 !* *
109 !* FNDBIT ( AMIN, AMAX, RDB, NBITS, ISCALE, IRET ) *
110 !* *
111 !* Input parameters: *
112 !* AMIN REAL Minimum value *
113 !* AMAX REAL Maximum value *
114 !* RDB REAL Maximum # of significant digits *
115 !* OR binary precision if < 0 *
116 !* *
117 !* Output parameters: *
118 !* NBITS INTEGER Number of bits for packing *
119 !* ISCALE INTEGER Power of 10 scaling to use *
120 !* IRET INTEGER Return code *
121 !* 0 = normal return *
122 !** *
123 !* Log: *
124 !* K. Brill/NMC 06/92 *
125 !* K. Brill/EMC 12/95 Added binary precision *
126 !* M. Baldwin 10/96 Added fix for negative nmbts
127 !************************************************************************
128 !*
129 !
130  implicit none
131 !
132  integer,intent(inout) :: iscale,nmbts
133  real,intent(inout) :: rmin,rmax,rdb
134  real :: range,rr,rng2,po,rln2
135  integer :: iret,icnt,ipo,le,ibin
136 !
137  DATA rln2/0.69314718/
138 !-----------------------------------------------------------------------
139  iret = 0
140  icnt = 0
141  iscale = 0
142  range = rmax - rmin
143  IF ( range .le. 0.00 ) THEN
144  nmbts = 8
145  RETURN
146  END IF
147 !*
148  IF ( rdb .eq. 0.0 ) THEN
149  nmbts = 8
150  RETURN
151  ELSE IF ( rdb .gt. 0.0 ) THEN
152  ipo = int(alog10( range ))
153  IF ( range .lt. 1.00 ) ipo = ipo - 1
154  po = float(ipo) - rdb + 1.
155  iscale = - int( po )
156  rr = range * 10. ** ( -po )
157  nmbts = int( alog( rr ) / rln2 ) + 1
158  ELSE
159  ibin = nint( -rdb )
160  rng2 = range * 2. ** ibin
161  nmbts = int( alog( rng2 ) / rln2 ) + 1
162  END IF
163 !*
164  IF(nmbts.LE.0) THEN
165  nmbts=0
166  IF(abs(rmin).GE.1.) THEN
167  iscale=-int(alog10(abs(rmin)))
168  ELSE IF (abs(rmin).LT.1.0.AND.abs(rmin).GT.0.0) THEN
169  iscale=-int(alog10(abs(rmin)))+1
170  ELSE
171  iscale=0
172  ENDIF
173  ENDIF
174  RETURN
175  END
