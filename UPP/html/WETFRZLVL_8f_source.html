2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: WETFRZLVL COMPUTES LEVEL OF 0 WET BULB
5 ! PRGRMMR: MANIKIN ORG: W/NP2 DATE: 03-11-14
6 !
7 ! ABSTRACT:
8 ! THIS ROUTINE COMPUTES THE LOWEST HEIGHT WITH A WET BULB
9 ! TEMPERATURE OF FREEZING FOR EACH MASS POINT ON THE ETA GRID.
10 ! THE COMPUTED WET BULB ZERO HEIGHT IS THE MEAN SEA LEVEL
11 ! HEIGHT. AT EACH MASS POINT WE MOVE UP FROM THE SURFACE TO
12 ! FIND THE FIRST ETA LAYER WHERE THE TW IS LESS THAN
13 ! 273.16K. VERTICAL INTERPOLATION IN TEMPERATURE TO THE FREEZING
14 ! TEMPERATURE GIVES THE FREEZING LEVEL HEIGHT. PRESSURE AND
15 ! SPECIFIC HUMIDITY ARE INTERPOLATED TO THIS LEVEL AND ALONG WITH
16 ! THE TEMPERATURE PROVIDE THE FREEZING LEVEL RELATIVE HUMIDITY.
17 ! IF THE SURFACE (SKIN) TEMPERATURE IS BELOW FREEZING, THE ROUTINE
18 ! USES SURFACE BASED FIELDS TO COMPUTE THE RELATIVE HUMIDITY.
19 !
20 ! PROGRAM HISTORY LOG:
21 ! 03-11-14 GEOFF MANIKIN - NEW PROGRAM
22 ! 04-12-06 G MANIKIN - CORRECTED COMPUTATION OF SFC TEMPERATURE
23 ! 05-03-11 H CHUANG - WRF VERSION
24 !
25 ! USAGE: CALL WETFRZLVL(TWET,ZWET)
26 ! INPUT ARGUMENT LIST:
27 ! TWET - WET BULB TEMPERATURES
28 !
29 ! OUTPUT ARGUMENT LIST:
30 ! ZWET - ABOVE GROUND LEVEL HEIGHT OF LEVEL WITH 0 WET BULB.
31 !
32 ! OUTPUT FILES:
33 ! NONE
34 !
35 ! SUBPROGRAMS CALLED:
36 ! UTILITIES:
37 ! REL_HUM
38 ! LIBRARY:
39 ! COMMON -
40 ! LOOPS
41 ! PVRBLS
42 ! MASKS
43 ! MAPOT
44 ! POSTVAR
45 !
46 ! ATTRIBUTES:
47 ! LANGUAGE: FORTRAN
48 ! MACHINE : CRAY C-90
49 !$$$
50 !
51 !
52  use vrbls3d, only: pint, zint, t
53  use vrbls2d, only: fis, thz0, ths
54  use masks, only: lmh, sm
55  use params_mod, only: gi, p1000, capa, tfrz, d0065, d50
56  use ctlblk_mod, only: jsta, jend, im, jsta_2l, jend_2u, lm
57 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
58  implicit none
59 !
60 ! DECLARE VARIABLES.
61 !
62  REAL,intent(in) :: TWET(IM,JSTA_2L:JEND_2U,LM)
63  REAL,intent(out) :: ZWET(IM,jsta:jend)
64 !
65  integer I,J,LLMH,L
66  real HTSFC,THSFC,PSFC,TSFC,DELZ,DELT,ZL,ZU
67 !*********************************************************************
68 ! START FRZLVL.
69 !
70 ! LOOP OVER HORIZONTAL GRID.
71 !
72 !!$omp parallel do
73 !!$omp& private(delt,delz,htsfc,l,llmh
74 !!$omp& tsfc,zl,zu)
75  DO 20 j=jsta,jend
76  DO 20 i=1,im
77  htsfc = fis(i,j)*gi
78  llmh = nint(lmh(i,j))
79  zwet(i,j) = htsfc
80 !
81 ! CHECK IF FREEZING LEVEL IS AT THE GROUND.
82 ! IF YES, ESTIMATE UNDERGROUND FREEZING LEVEL USING 6.5C/KM LAPSE RATE
83 ! AND ASSUME RH TO BE EQUAL TO RH AT SFC
84 !
85  thsfc = (sm(i,j)*thz0(i,j)+(1.-sm(i,j))*ths(i,j))
86  psfc = pint(i,j,llmh+1)
87  tsfc = thsfc*(psfc/p1000)**capa
88 
89  IF (tsfc.LE.tfrz) THEN
90 ! ZWET(I,J) = HTSFC
91  zwet(i,j) = htsfc+(tsfc-tfrz)/d0065
92  GOTO 20
93  ENDIF
94 !
95 ! OTHERWISE, LOCATE THE FREEZING LEVEL ALOFT.
96 !
97  DO 10 l = llmh,1,-1
98  IF (twet(i,j,l).LE.tfrz) THEN
99  IF (l.LT.llmh-1) THEN
100  delz = d50*(zint(i,j,l)-zint(i,j,l+2))
101  zl = d50*(zint(i,j,l+1)+zint(i,j,l+2))
102  delt = twet(i,j,l)-twet(i,j,l+1)
103  zwet(i,j) = zl + (tfrz-twet(i,j,l+1))/delt*delz
104  ELSE
105  zu = d50*(zint(i,j,l)+zint(i,j,l+1))
106  zl = htsfc
107  delz = zu-zl
108  tsfc = sm(i,j)*thz0(i,j)+(1.-sm(i,j))*ths(i,j) &
109  *(pint(i,j,nint(lmh(i,j))+1)/p1000)**capa
110  delt = t(i,j,l)-tsfc
111  IF(delt .NE. 0.)THEN
112  zwet(i,j) = zl + (tfrz-tsfc)/delt*delz
113  ELSE
114  zwet(i,j) = htsfc+(tsfc-twet(i,j,l))/d0065
115  END IF
116  IF (zwet(i,j) .GT. zu) THEN
117  zwet(i,j)=zu
118  ENDIF
119  IF ((-1*zwet(i,j)) .GT. zu) THEN
120  zwet(i,j)=zu
121  endif
122  ENDIF
123  GOTO 20
124  ENDIF
125  10 CONTINUE
126  20 CONTINUE
127 !
128 ! END OF ROUTINE.
129 !
130  RETURN
131  END
