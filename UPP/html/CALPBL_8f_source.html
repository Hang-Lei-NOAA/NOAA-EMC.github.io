88 !!$omp parallel do
89 !!$omp& private(uhkl,ulkl,vhkl,vlkl,rib,ubot,utop,vbot,vtop,

129 !

133  ENDIF
134 
135  IF(gridtype == 'A') THEN

137  utop = uh(i,j,l-1)
138  vbot = vh(i,j,l)
139  vtop = vh(i,j,l-1)
140  ELSE IF(gridtype == 'E') THEN
141  ie = i+mod(j+1,2)
142  iw = i+mod(j+1,2)-1
143 !
144 ! WE NEED (U,V) WINDS AT A MASS POINT. FOUR POINT
145 ! AVERAGE (U,V) WINDS TO MASS POINT. NORMALIZE FOUR
146 ! POINT AVERAGE BY THE ACTUAL NUMBER OF (U,V) WINDS
147 ! USED IN THE AVERAGING. VTM=1 IF WIND POINT IS

149 !
150  wndsl = vtm(i,j-1,l)+vtm(iw,j,l)+vtm(ie,j,l)+vtm(i,j+1,l)

153  IF(wndsl == 0. .OR. wndslp == 0.) cycle
154  ubot = (uh(i,j-1,l)+uh(iw,j,l)+uh(ie,j,l)+uh(i,j+1,l))/wndsl
155  utop = (uh(i,j-1,l-1)+uh(iw,j,l-1)+uh(ie,j,l-1)+ &

166  vbot = (vh(iw,j-1,l)+vh(iw,j,l)+vh(ie,j-1,l)+vh(i,j,l))*0.25

169  END IF

177  ENDIF
178 
179  thvtop = thv(i,j,l-1)

182 !
183 ! COMPUTE BULK RICHARDSON NUMBER.
184 !
185 ! FOLLOWING VOGELEZANG AND HOLTSLAG (1996):
186 
187  wdl2 = (utop-ubot1(i,j))**2 + (vtop-vbot1(i,j))**2 + wmin**2
188  rib = (g/thvbot1(i,j))*(thvtop-thvbot1(i,j))* &
189  (ztop-zbot1(i,j))/(wdl2+betta*(ustarr**2))
190 !
191 ! COMPUTE PBL HEIGHT
192 !
193 ! --------------------------------------------------------------------
194 ! IF BULK RICHARDSON NUMBER (RIB) EXCEEDS THE CRITICAL RICHARDSON

196 ! BETWEEN HEIGHTS, AND PREVIOUS (RIBP) AND CURRENT (RIB) BULK

198 ! --------------------------------------------------------------------

201  (ricr-ribp(i,j))/(rib-ribp(i,j))
202  icalpbl(i,j) = 1
203 
204 !-------- Extract surface height -----------------------------------
205 
206  pblri(i,j) = pblri(i,j)-fis(i,j)*gi
207 
208  ENDIF
209 
210  ribp(i,j) = rib
211  lvlp(i,j) = l-1
212 !

214  ENDDO
215  ENDDO
216  ENDDO
