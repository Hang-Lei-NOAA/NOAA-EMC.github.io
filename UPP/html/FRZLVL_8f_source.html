1  SUBROUTINE frzlvl(ZFRZ,RHFRZ,PFRZL)
2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: FRZLVL COMPUTES FRZING LVL Z AND RH
5 ! PRGRMMR: TREADON ORG: W/NP2 DATE: 92-12-22
6 ! T. Smirnova - 8-27-2010 - added PFRZL to the output
7 !
8 ! ABSTRACT:
9 ! THIS ROUTINE COMPUTES THE FREEZING LEVEL HEIGHT AND RELATIVE
10 ! HUMIDITY AT THIS LEVEL FOR EACH MASS POINT ON THE ETA GRID.
11 ! THE COMPUTED FREEZING LEVEL HEIGHT IS THE MEAN SEA LEVEL
12 ! HEIGHT. AT EACH MASS POINT WE MOVE UP FROM THE SURFACE TO
13 ! FIND THE FIRST ETA LAYER WHERE THE TEMPERATURE IS LESS THAN
14 ! 273.16K. VERTICAL INTERPOLATION IN TEMPERATURE TO THE FREEZING
15 ! TEMPERATURE GIVES THE FREEZING LEVEL HEIGHT. PRESSURE AND
16 ! SPECIFIC HUMIDITY ARE INTERPOLATED TO THIS LEVEL AND ALONG WITH
17 ! THE TEMPERATURE PROVIDE THE FREEZING LEVEL RELATIVE HUMIDITY.
18 ! IF THE SURFACE (SKIN) TEMPERATURE IS BELOW FREEZING, THE ROUTINE
19 ! USES SURFACE BASED FIELDS TO COMPUTE THE RELATIVE HUMIDITY.
20 !
21 ! NOTE THAT IN POSTING FREEZING LEVEL DATA THE LFM LOOK-ALIKE FILE
22 ! (IE, GRID 26), WE PACK 273.15K AS THE FREEZING TEMPERATURE. ALL
23 ! OTHER OUTPUT GRIDS USE 273.16K
24 ! .
25 !
26 ! PROGRAM HISTORY LOG:
27 ! 92-12-22 RUSS TREADON
28 ! 93-06-10 RUSS TREADON - CORRECTED FREEZING LEVEL HEIGHTS TO BE
29 ! WITH REPSECT TO MEAN SEA LEVEL, NOT
30 ! ABOVE GROUND LEVEL.
31 ! 98-06-15 T BLACK - CONVERSION FROM 1-D TO 2-D
32 ! 98-08-17 MIKE BALDWIN - COMPUTE RH OVER ICE IF NECESSARY
33 ! 98-12-22 MIKE BALDWIN - BACK OUT RH OVER ICE
34 ! 00-01-04 JIM TUCCILLO - MPI VERSION
35 ! 01-10-25 H CHUANG - MODIFIED TO PROCESS HYBRID MODEL OUTPUT
36 ! 02-01-15 MIKE BALDWIN - WRF VERSION
37 !
38 ! USAGE: CALL FRZLVL(ZFRZ,RHFRZ)
39 ! INPUT ARGUMENT LIST:
40 ! NONE
41 !
42 ! OUTPUT ARGUMENT LIST:
43 ! ZFRZ - ABOVE GROUND LEVEL FREEZING HEIGHT.
44 ! RHFRZ - RELATIVE HUMIDITY AT FREEZING LEVEL.
45 ! PFRZL - PRESSURE AT FREEZING LEVEL.
46 !
47 ! OUTPUT FILES:
48 ! NONE
49 !
50 ! SUBPROGRAMS CALLED:
51 ! UTILITIES:
52 ! NONE
53 ! LIBRARY:
54 ! COMMON -
55 ! LOOPS
56 ! PVRBLS
57 ! MASKS
58 ! MAPOT
59 ! POSTVAR
60 !
61 ! ATTRIBUTES:
62 ! LANGUAGE: FORTRAN
63 ! MACHINE : CRAY C-90
64 !$$$
65 !
66 !
67  use vrbls3d, only: pint, t, zmid, q, pmid
68  use vrbls2d, only: fis, tshltr, pshltr, qshltr
69  use masks, only: lmh
70  use params_mod, only: gi, d00, capa, d0065, tfrz, pq0, a2, a3, a4
71  use ctlblk_mod, only: jsta, jend, spval, lm, modelname, im
72  use physcons_post, only: con_rd, con_rv, con_eps, con_epsm1
73 
74  implicit none
75 
76  real,external::FPVSNEW
77 !
78 ! implicit none
79 !
80 ! DECLARE VARIABLES.
81 !
82  REAL,dimension(im,jsta:jend) :: RHFRZ, ZFRZ, PFRZL
83  integer I,J,LLMH,L
84  real HTSFC,PSFC,TSFC,QSFC,QSAT,RHSFC,DELZ,DELT,DELQ,DELALP, &
85  delzp,zl,dzabv,qfrz,alpl,alph,alpfrz,pfrz,qsfrz,rhz,zu, &
86  dzfr,es
87 !
88 !*********************************************************************
89 ! START FRZLVL.
90 !
91 !
92 !
93 ! LOOP OVER HORIZONTAL GRID.
94 !
95 !!$omp parallel do &
96 ! & private(i,j,alpfrz,alph,alpl,delalp,delq,delt,delz, &
97 ! & delzp,dzabv,dzfr,htsfc,l,llmh,psfc,qfrz, &
98 ! & qsat,qsfc,qsfrz,rhsfc,rhz,tsfc, &
99 ! & zl,zu)
100  DO 20 j=jsta,jend
101  DO 20 i=1,im
102  htsfc = fis(i,j)*gi
103  llmh = nint(lmh(i,j))
104  rhfrz(i,j) = d00
105  zfrz(i,j) = htsfc
106  psfc = pint(i,j,llmh+1)
107  pfrzl(i,j) = psfc
108 !
109 ! CHECK IF FREEZING LEVEL IS AT THE GROUND.
110 !
111 ! IF(SM(I,J)/=SPVAL .AND. THZ0(I,J)/=SPVAL .AND. &
112 ! THS(I,J)/=SPVAL)THEN
113 ! TSFC = (SM(I,J)*THZ0(I,J)+(1.-SM(I,J))*THS(I,J)) &
114 ! *(PINT(I,J,NINT(LMH(I,J))+1)/P1000)**CAPA
115 ! Per AWC's request, use 2m T instead of skin T so that freezing level
116 ! would be above ground more often
117  IF(tshltr(i,j)/=spval .AND. pshltr(i,j)/=spval)THEN
118  tsfc=tshltr(i,j)*(pshltr(i,j)*1.e-5)**capa
119  ELSE
120 ! GFS analysis does not have flux file to retrieve TSFC from
121  tsfc=t(i,j,lm)+d0065*(zmid(i,j,lm)-htsfc-2.0)
122  END IF
123  IF (tsfc.LE.tfrz) THEN
124 ! ZFRZ(I,J) = HTSFC+(TSFC-TFRZ)/D0065
125  zfrz(i,j) = htsfc+2.0+(tsfc-tfrz)/d0065
126 ! IF(SM(I,J)/=SPVAL .AND. QZ0(I,J)/=SPVAL .AND. &
127 ! QS(I,J)/=SPVAL)THEN
128 ! QSFC = SM(I,J)*QZ0(I,J)+(1.-SM(I,J))*QS(I,J)
129 ! GFS does not output QS
130 ! ELSE IF(QSHLTR(I,J)/=SPVAL)THEN
131  IF(qshltr(i,j)/=spval)THEN
132  psfc=pshltr(i,j)
133  qsfc=qshltr(i,j)
134  ELSE
135  qsfc=q(i,j,lm)
136  psfc=pmid(i,j,lm)
137  END IF
138 !
139  IF(modelname == 'GFS' .OR. modelname == 'RAPR')THEN
140  es=fpvsnew(tsfc)
141  es=min(es,psfc)
142  qsat=con_eps*es/(psfc+con_epsm1*es)
143  ELSE
144  qsat=pq0/psfc*exp(a2*(tsfc-a3)/(tsfc-a4))
145  END IF
146 !
147  rhsfc = qsfc/qsat
148  rhsfc = amax1(0.01,rhsfc)
149  rhsfc = amin1(rhsfc,1.0)
150  rhfrz(i,j)= rhsfc
151  pfrzl(i,j)= psfc
152  GOTO 20
153  ENDIF
154 !
155 ! OTHERWISE, LOCATE THE FREEZING LEVEL ALOFT.
156 !
157  DO 10 l = llmh,1,-1
158  IF (t(i,j,l).LE.tfrz) THEN
159  IF (l.LT.llmh) THEN
160  delz = zmid(i,j,l)-zmid(i,j,l+1)
161  zl = zmid(i,j,l+1)
162  delt = t(i,j,l)-t(i,j,l+1)
163  zfrz(i,j) = zl + (tfrz-t(i,j,l+1))/delt*delz
164 !
165  dzabv = zfrz(i,j)-zl
166  delq = q(i,j,l)-q(i,j,l+1)
167  qfrz = q(i,j,l+1) + delq/delz*dzabv
168  qfrz = amax1(0.0,qfrz)
169 !
170 !
171  alpl = alog(pmid(i,j,l+1))
172  alph = alog(pmid(i,j,l))
173  alpfrz = alpl + (alph-alpl)/delz*dzabv
174  pfrz = exp(alpfrz)
175  pfrzl(i,j) = pfrz
176  IF(modelname == 'GFS' .OR.modelname == 'RAPR')THEN
177  es=fpvsnew(tfrz)
178  es=min(es,pfrz)
179  qsfrz=con_eps*es/(pfrz+con_epsm1*es)
180  ELSE
181  qsfrz=pq0/pfrz &
182  *exp(a2*(tfrz-a3)/(tfrz-a4))
183  END IF
184 !
185  rhz = qfrz/qsfrz
186  rhz = amax1(0.01,rhz)
187  rhz = amin1(rhz,1.0)
188  rhfrz(i,j) = rhz
189 !
190  ELSE
191  zu = zmid(i,j,l)
192  zl = htsfc+2.0
193  delz = zu-zl
194  IF(tshltr(i,j)/=spval .AND. pshltr(i,j)/=spval)THEN
195  tsfc=tshltr(i,j)*(pshltr(i,j)*1.e-5)**capa
196  ELSE
197 ! GFS analysis does not have flux file to retrieve TSFC from
198  tsfc=t(i,j,lm)+d0065*(zmid(i,j,lm)-htsfc-2.0)
199  END IF
200  delt = t(i,j,l)-tsfc
201  zfrz(i,j) = zl + (tfrz-tsfc)/delt*delz
202 !
203  dzabv = zfrz(i,j)-zl
204 ! GFS does not output QS
205  IF(qshltr(i,j)/=spval)THEN
206  qsfc=qshltr(i,j)
207  ELSE
208  qsfc=q(i,j,lm)
209  END IF
210  delq = q(i,j,l)-qsfc
211  qfrz = qsfc + delq/delz*dzabv
212  qfrz = amax1(0.0,qfrz)
213 !
214  alph = alog(pmid(i,j,l))
215  alpl = alog(psfc)
216  delalp = alph-alpl
217  alpfrz = alpl + delalp/delz*dzabv
218  pfrz = exp(alpfrz)
219 !
220  pfrzl(i,j) = pfrz
221  IF(modelname == 'GFS'.OR.modelname == 'RAPR')THEN
222  es=fpvsnew(tfrz)
223  es=min(es,pfrz)
224  qsfrz=con_eps*es/(pfrz+con_epsm1*es)
225  ELSE
226  qsfrz=pq0/pfrz &
227  *exp(a2*(tfrz-a3)/(tfrz-a4))
228  END IF
229 !
230  rhz = qfrz/qsfrz
231  rhz = amax1(0.01,rhz)
232  rhz = amin1(rhz,1.0)
233  rhfrz(i,j)= rhz
234  ENDIF
235 !
236 ! BOUND FREEZING LEVEL RH. FREEZING LEVEL HEIGHT IS
237 ! MEASURED WITH RESPECT TO MEAN SEA LEVEL.
238 !
239 ! RHFRZ(I,J) = AMAX1(0.01,RHFRZ(I,J))
240 ! RHFRZ(I,J) = AMIN1(RHFRZ(I,J),1.00)
241  zfrz(i,j) = amax1(0.0,zfrz(i,j))
242  GOTO 20
243  ENDIF
244  10 CONTINUE
245  20 CONTINUE
246 !
247 ! END OF ROUTINE.
248 !
249  RETURN
250  END
