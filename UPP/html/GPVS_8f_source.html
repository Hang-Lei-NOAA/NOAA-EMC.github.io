12 ! ABSTRACT: COMPUTE SATURATION VAPOR PRESSURE TABLE AS A FUNCTION OF

70 ! AUTHOR: N PHILLIPS W/NP2 DATE: 30 DEC 82

72 ! ABSTRACT: COMPUTE SATURATION VAPOR PRESSURE FROM THE TEMPERATURE.
73 ! A LINEAR INTERPOLATION IS DONE BETWEEN VALUES IN A LOOKUP TABLE

78 ! THIS FUNCTION SHOULD BE EXPANDED INLINE IN THE CALLING ROUTINE.

144  FUNCTION fpvsx(T)
145 !-----------------------------------------------------------------------
146 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
147 ! . . .
148 ! SUBPROGRAM: FPVSX COMPUTE SATURATION VAPOR PRESSURE
149 ! AUTHOR: N PHILLIPS W/NP2 DATE: 30 DEC 82
150 !
151 ! ABSTRACT: EXACTLY COMPUTE SATURATION VAPOR PRESSURE FROM TEMPERATURE.
152 ! THE WATER MODEL ASSUMES A PERFECT GAS, CONSTANT SPECIFIC HEATS
153 ! FOR GAS AND LIQUID, AND NEGLECTS THE VOLUME OF THE LIQUID.
154 ! THE MODEL DOES ACCOUNT FOR THE VARIATION OF THE LATENT HEAT
155 ! OF CONDENSATION WITH TEMPERATURE. THE ICE OPTION IS NOT INCLUDED.
156 ! THE CLAUSIUS-CLAPEYRON EQUATION IS INTEGRATED FROM THE TRIPLE POINT
157 ! TO GET THE FORMULA
158 ! PVS=PSATK*(TR**XA)*EXP(XB*(1.-TR))
159 ! WHERE TR IS TTP/T AND OTHER VALUES ARE PHYSICAL CONSTANTS
160 ! THIS FUNCTION SHOULD BE EXPANDED INLINE IN THE CALLING ROUTINE.
161 !
162 ! PROGRAM HISTORY LOG:
163 ! 91-05-07 IREDELL MADE INTO INLINABLE FUNCTION
164 ! 94-12-30 IREDELL EXACT COMPUTATION
165 ! 96-02-19 HONG ICE EFFECT
166 !
167 ! USAGE: PVS=FPVSX(T)
168 ! REFERENCE: EMANUEL(1994),116-117
169 !
170 ! INPUT ARGUMENT LIST:
171 ! T - REAL TEMPERATURE IN KELVIN
172 !
173 ! OUTPUT ARGUMENT LIST:
174 ! FPVSX - REAL SATURATION VAPOR PRESSURE IN KILOPASCALS (CB)
175 !
176 ! ATTRIBUTES:
177 ! LANGUAGE: FORTRAN 90
178 ! MACHINE: IBM SP
179 !
180 !$$$
181 !-----------------------------------------------------------------------
182  implicit none
183 !
184  real,PARAMETER :: cp=1.0046e+3,rd=287.04,rv=4.6150e+2, &
185  ttp=2.7316e+2,hvap=2.5000e+6,psat=6.1078e+2, &
186  cliq=4.1855e+3,cvap= 1.8460e+3,cice=2.1060e+3,hsub=2.8340e+6
187  real,PARAMETER :: psatk=psat*1.e-3
188  real,PARAMETER :: dldt=cvap-cliq,xa=-dldt/rv,xb=xa+hvap/(rv*ttp)
189  real,PARAMETER :: dldti=cvap-cice,xai=-dldti/rv,xbi=xai+hsub/(rv*ttp)
190  real :: tr, t
191  real :: fpvsx
192 !-----------------------------------------------------------------------
193  tr=ttp/t
194 !
195  IF(t.GE.ttp)THEN
196  fpvsx=psatk*(tr**xa)*exp(xb*(1.-tr))
197  ELSE
198  fpvsx=psatk*(tr**xai)*exp(xbi*(1.-tr))
199  ENDIF
200 !
201  RETURN
202  END
